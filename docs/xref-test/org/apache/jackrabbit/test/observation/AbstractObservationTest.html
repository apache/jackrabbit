43      protected void setUp() throws Exception {
44  	super.setUp();
45          obsMgr = superuser.getWorkspace().getObservationManager();

48      /***
49       * Registers an <code>EventListener</code> for all events.
50       *
51       * @param listener the <code>EventListener</code>.
52       * @throws RepositoryException if registration fails.
53       */
54      protected void addEventListener(EventListener listener) throws RepositoryException {
55          addEventListener(listener,
56                  EventType.CHILD_NODE_ADDED | EventType.CHILD_NODE_REMOVED | EventType.PROPERTY_ADDED | EventType.PROPERTY_CHANGED | EventType.PROPERTY_REMOVED);
57      }
58  
59      /***
60       * Registers an <code>EventListener</code> for events of the specified
61       * type(s).
62       *
63       * @param listener the <code>EventListener</code>.
64       * @param eventType the {@link javax.jcr.observation.EventType}s
65       * @throws RepositoryException if registration fails.
66       */
67      protected void addEventListener(EventListener listener, long eventType)
68              throws RepositoryException {
69          if (obsMgr != null) {
70              obsMgr.addEventListener(listener,
71                      eventType,
72                      superuser.getRootNode().getPath(),
73                      true,
74                      null,
75                      null,
76                      false);
77          } else {
78              throw new IllegalStateException("ObservationManager not available.");
79          }
80      }
81  
82      /***
83       * Removes the <code>EventListener</code> from the ObservationManager.
84       *
85       * @param listener the <code>EventListener</code> to unregister.
86       * @throws RepositoryException if unregister fails.
87       */
88      protected void removeEventListener(EventListener listener) throws RepositoryException {
89          if (obsMgr != null) {
90              obsMgr.removeEventListener(listener);
91          } else {
92              throw new IllegalStateException("ObservationManager not available.");
93          }
94      }

173     /***
174      * Checks <code>Events</code> for paths. All <code>relPaths</code> are
175      * relative to {@link #TEST_ROOT}.
176      *
177      * @param events    the <code>Event</code>s.
178      * @param relPaths  paths to item events relative to {@link #TEST_ROOT}.
179      * @param eventType the type of event to check.
180      * @throws RepositoryException if an error occurs while retrieving the nodes
181      *                             from event instances.
182      */
183     private void checkNodes(Event[] events, String[] relPaths, long eventType)
184 	    throws RepositoryException {
185 	assertEquals("Number of events wrong", relPaths.length, events.length);
186 	Set paths = new HashSet();
187 	for (int i = 0; i < events.length; i++) {
188 	    assertEquals("Wrong event type", eventType, events[i].getType());
189 	    String path = events[i].getNodePath()
190 		    + "/" + events[i].getChildName();
191 	    paths.add(path);
192 	}
193 	for (int i = 0; i < relPaths.length; i++) {
194 	    String expected = "/" + TEST_ROOT + "/" + relPaths[i];
195 	    assertTrue("Path " + expected + " not found in events.",
196 		    paths.contains(expected));
197 	}
198     }
