32      public void testSingleNodeAdded() throws RepositoryException {
33          EventResult result = new EventResult(log);
34          addEventListener(result, EventType.CHILD_NODE_ADDED);
35          testRoot.addNode("foo", NT_UNSTRUCTURED);
36          testRoot.save();
37          removeEventListener(result);
38          Event[] events = result.getEvents(DEFAULT_WAIT_TIMEOUT);
39          checkNodeAdded(events, new String[] { "foo" });
40      }
