16  package org.apache.jackrabbit.test;

18  import javax.jcr.Session;
19  import javax.jcr.Node;
20  
21  /***
22   * @author Marcel Reutegger
23   * @version $Revision:  $, $Date:  $
24   */
25  public abstract class AbstractTest extends JUnitTest {
26  
27      protected static final String TEST_ROOT = "testroot";
28  
29      protected static final String JCR_PRIMARY_TYPE = "jcr:primaryType";
30  
31      protected static final String NT_UNSTRUCTURED = "nt:unstructured";
32  
33      protected static final String MIX_REFERENCABLE = "mix:referencable";
34  
35      protected static final String NT_BASE = "nt:base";
36  
37      /*** The superuser session */
38      protected Session superuser;
39  
40      /*** The root <code>Node</code> for testing */
41      protected Node testRoot;
42  
43      protected void setUp() throws Exception {
44          superuser = helper.getSuperuserSession();
45          Node root = superuser.getRootNode();
46          if (root.hasNode(TEST_ROOT)) {
47              // remove test root
48              root.remove(TEST_ROOT);
49              root.save();
50          }
51          testRoot = root.addNode(TEST_ROOT, NT_UNSTRUCTURED);
52          root.save();
53      }
54  
55      protected void tearDown() throws Exception {
56          if (superuser != null) {
57              // do a 'rollback'
58              superuser.refresh(false);
59              Node root = superuser.getRootNode();
60              if (root.hasNode(TEST_ROOT)) {
61                  root.remove(TEST_ROOT);
62                  root.save();
63              }
64              superuser.logout();
65          }
66      }
67  
68  }
