16  package org.apache.jackrabbit.core.observation;
17  
18  import org.apache.log4j.Logger;
19  
20  import javax.jcr.RepositoryException;
21  import javax.jcr.observation.Event;
22  import javax.jcr.observation.EventIterator;
23  import java.util.Iterator;
24  import java.util.NoSuchElementException;
25  import java.util.Set;
26  
27  /***
28   * @author Marcel Reutegger
29   */
30  class FilteredEventIterator implements EventIterator {
31  
32      /***
33       * Logger instance for this class
34       */
35      private static final Logger log
36  	    = Logger.getLogger(FilteredEventIterator.class);
37  
38      /***
39       * The actual {@link EventState}s fired by the workspace (unfiltered)
40       */
41      private final Iterator actualEvents;
42  
43      /***
44       * For filtering the {@link javax.jcr.observation.Event}s.
45       */
46      private final EventFilter filter;
47  
48      private final Set denied;
49  
50      /***
51       * The next {@link javax.jcr.observation.Event} in this iterator
52       */
53      private Event next;
54  
55      /***
56       * Current position
57       */
58      private long pos = 0;
59  
60      /***
61       * Creates a new <code>FilteredEventIterator</code>.
62       *
63       * @param c      an unmodifiable Collection of {@link javax.jcr.observation.Event}s.
64       * @param filter only event that pass the filter will be dispatched to the
65       *               event listener.
66       * @param denied <code>Set</code> of denied <code>EventState</code>s
67       *               rejected by the <code>AccessManager</code>. If
68       *               <code>null</code> no <code>EventState</code> is denied.
69       */
70      public FilteredEventIterator(EventStateCollection c,
71  				 EventFilter filter,
72  				 Set denied) {
73  	actualEvents = c.iterator();
74  	this.filter = filter;
75  	this.denied = denied;
76  	fetchNext();
77      }
78  
79      /***
80       * @see Iterator#next()
81       */
82      public Object next() {
83  	if (next == null) {
84  	    throw new NoSuchElementException();
85  	}
86  	Event e = next;
87  	fetchNext();
88  	pos++;
89  	return e;
90      }
91  
92      /***
93       * @see EventIterator#nextEvent()
94       */
95      public Event nextEvent() {
96  	return (Event) next();
97      }
98  
99      /***
100      * @see javax.jcr.RangeIterator#skip(long)
101      */
102     public void skip(long skipNum) {
103 	while (skipNum-- > 0) {
104 	    next();
105 	}
106     }
107 
108     /***
109      * Always returns <code>-1</code>.
110      *
111      * @return <code>-1</code>.
112      */
113     public long getSize() {
114 	return -1;
115     }
116 
117     /***
118      * @see javax.jcr.RangeIterator#getPos()
119      */
120     public long getPos() {
121 	return pos;
122     }
123 
124     /***
125      * This method is not supported.
126      * Always throws a <code>UnsupportedOperationException</code>.
127      */
128     public void remove() {
129 	throw new UnsupportedOperationException("EventIterator.remove()");
130     }
131 
132     /***
133      * Returns <tt>true</tt> if the iteration has more elements. (In other
134      * words, returns <tt>true</tt> if <tt>next</tt> would return an element
135      * rather than throwing an exception.)
136      *
137      * @return <tt>true</tt> if the iterator has more elements.
138      */
139     public boolean hasNext() {
140 	return (next != null);
141     }
142 
143     /***
144      * Fetches the next Event from the collection of events
145      * passed in the constructor of <code>FilteredEventIterator</code>
146      * that is allowed by the {@link EventFilter}.
147      */
148     private void fetchNext() {
149 	EventState state;
150 	next = null;
151 	while (next == null && actualEvents.hasNext()) {
152 	    state = (EventState) actualEvents.next();
153 	    // check denied set
154 	    if (denied == null || !denied.contains(state)) {
155 		try {
156 		    next = filter.blocks(state) ? null : new EventImpl(filter.getSession(),
157 			    /* filter.getItemManager(), */
158 			    state);
159 		} catch (RepositoryException e) {
160 		    log.error("Exception while applying filter.", e);
161 		}
162 	    }
163 	}
164     }
165 }
