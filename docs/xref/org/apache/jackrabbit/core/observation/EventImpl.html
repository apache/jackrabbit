16  package org.apache.jackrabbit.core.observation;
17  
18  import org.apache.log4j.Logger;
19  import org.apache.jackrabbit.core.ItemManager;
20  import org.apache.jackrabbit.core.NoPrefixDeclaredException;
21  import org.apache.jackrabbit.core.NodeId;
22  import org.apache.jackrabbit.core.SessionImpl;
23  
24  import javax.jcr.RepositoryException;
25  import javax.jcr.observation.Event;
26  
27  /***
28   * Implementation of the {@link javax.jcr.observation.Event} interface.
29   *
30   * @author Marcel Reutegger
31   * @version $Revision: 1.6 $
32   */
33  final class EventImpl implements Event {
34  
35      /***
36       * Logger instance for this class
37       */
38      private static final Logger log = Logger.getLogger(EventImpl.class);
39  
40      /***
41       * The session of the {@link javax.jcr.observation.EventListener} this
42       * event will be delivered to.
43       */
44      private final SessionImpl session;
45  
46      /***
47       * The <code>ItemManager</code> of the session.
48       */
49      private final ItemManager itemMgr;
50  
51      /***
52       * The shared {@link EventState} object.
53       */
54      private final EventState eventState;
55  
56      /***
57       * Cached String value of this <code>Event</code> instance.
58       */
59      private String stringValue;
60  
61      /***
62       * Creates a new {@link javax.jcr.observation.Event} instance based on an
63       * {@link EventState eventState}.
64       *
65       * @param session    the session of the registerd <code>EventListener</code>
66       *                   where this <code>Event</code> will be delivered to.
67       * @param itemMgr    the <code>ItemManager</code> of the above
68       *                   <code>Session</code>.
69       * @param eventState the underlying <code>EventState</code>.
70       */
71      EventImpl(SessionImpl session, ItemManager itemMgr, EventState eventState) {
72  	this.session = session;
73  	this.itemMgr = itemMgr;
74  	this.eventState = eventState;
75      }
76  
77      /***
78       * @see Event#getType()
79       */
80      public long getType() {
81  	return eventState.getType();
82      }
83  
84      /***
85       * @see Event#getNodePath()
86       */
87      public String getNodePath() throws RepositoryException {
88  	return itemMgr.getItem(new NodeId(eventState.getParentUUID())).getPath();
89      }
90  
91      /***
92       * @see Event#getChildName()
93       */
94      public String getChildName() throws RepositoryException {
95  	try {
96  	    return eventState.getChildItemQName().toJCRName(session.getNamespaceResolver());
97  	} catch (NoPrefixDeclaredException npde) {
98  	    // should never get here...
99  	    String msg = "internal error: encountered unregistered namespace in name";
100 	    log.error(msg, npde);
101 	    throw new RepositoryException(msg, npde);
102 	}
103     }
104 
105     /***
106      * @see Event#getUserId()
107      */
108     public String getUserId() {
109 	return eventState.getUserId();
110     }
111 
112     /***
113      * Returns a String representation of this <code>Event</code>.
114      *
115      * @return a String representation of this <code>Event</code>.
116      */
117     public String toString() {
118 	if (stringValue == null) {
119 	    StringBuffer sb = new StringBuffer();
120 	    sb.append("Event: Path: ");
121 	    try {
122 		sb.append(getNodePath());
123 	    } catch (RepositoryException e) {
124 		log.error("Exception retrieving path: " + e);
125 		sb.append("[Error retrieving path]");
126 	    }
127 	    sb.append(", ").append(EventState.valueOf(getType())).append(": ");
128 	    try {
129 		sb.append(getChildName());
130 	    } catch (RepositoryException e) {
131 		log.error("Exception retrieving child item name: " + e);
132 		sb.append("[Error retrieving child item name]");
133 	    }
134 	    sb.append(", UserId: ").append(getUserId());
135 	    stringValue = sb.toString();
136 	}
137 	return stringValue;
138     }
139 
140     /***
141      * @see Object#hashCode()
142      */
143     public int hashCode() {
144 	return eventState.hashCode() ^ session.hashCode();
145     }
146 
147     /***
148      * Returns <code>true</code> if this <code>Event</code> is equal to another
149      * object.
150      * <p/>
151      * Two <code>Event</code> instances are equal if their respective
152      * <code>EventState</code> instances are equal and both <code>Event</code>
153      * instances are intended for the same <code>Session</code> that registerd
154      * the <code>EventListener</code>.
155      *
156      * @param obj the reference object with which to compare.
157      * @return <code>true</code> if this <code>Event</code> is equal to another
158      *         object.
159      */
160     public boolean equals(Object obj) {
161 	if (this == obj) {
162 	    return true;
163 	}
164 	if (obj instanceof EventImpl) {
165 	    EventImpl other = (EventImpl) obj;
166 	    return this.eventState.equals(other.eventState)
167 		    && this.session.equals(other.session);
168 	}
169 	return false;
170     }
171 }
