32  /***
33   * The <code>EventConsumer</code> class combines the {@link
34   * javax.jcr.observation.EventListener} with the implementation of specified
35   * filter for the listener: {@link EventFilter}.

37   * Collections of {@link EventState} objects will be dispatched to {@link
38   * #consumeEvents}.

77      /***
78       * An <code>EventConsumer</code> consists of a <code>Session</code>, the
79       * attached <code>EventListener</code> and an <code>EventFilter</code>.
80       *
81       * @param session  the <code>Session</code> that created this
82       *                 <code>EventConsumer</code>.
83       * @param listener the actual <code>EventListener</code> to call back.
84       * @param filter   only pass an <code>Event</code> to the listener if the
85       *                 <code>EventFilter</code> allows the <code>Event</code>.
86       * @throws NullPointerException if <code>session</code>, <code>listener</code>
87       *                              or <code>filter</code> is<code>null</code>.
88       */
89      EventConsumer(SessionImpl session, EventListener listener, EventFilter filter) {
90  	if (session == null) {
91  	    throw new NullPointerException("session");
92  	}
93  	if (listener == null) {
94  	    throw new NullPointerException("listener");
95  	}
96  	if (filter == null) {
97  	    throw new NullPointerException("filter");
98  	}
99  
100 	this.session = session;
101 	this.listener = listener;
102 	this.filter = filter;
103     }

125     /***
126      * Checks for what {@link EventState}s this <code>EventConsumer</code> has
127      * enough access rights to see the event.
128      *
129      * @param events the collection of {@link EventState}s.
130      * @throws RepositoryException if an error occurs while checking access
131      *   rights on the {@link EventStateCollection}.
132      */
133     void prepareEvents(EventStateCollection events) throws RepositoryException {
134 	Iterator it = events.iterator();
135 	Set denied = null;
136 	while (it.hasNext()) {
137 	    EventState state = (EventState)it.next();
138 	    if (state.getType() == EventType.CHILD_NODE_REMOVED
139 		    || state.getType() == EventType.PROPERTY_REMOVED) {
140 
141 		if (session.equals(state.getSession())) {
142 		    // if we created the event, we can be sure that
143 		    // we have enough access rights to see the event
144 		    continue;
145 		}
146 
147 		try {
148 		    // FIXME is there a better way to check access right
149 		    // without transforming path
150 		    Path p = Path.create(state.getParentPath(), state.getChildItemQName(), false);
151 		    if (!session.getWorkspace().getAccessManager().isGranted(
152 			    p.toJCRPath(session.getNamespaceResolver()),
153 			    Permission.READ_ITEM)) {
154 			if (denied == null) {
155 			    denied = new HashSet();
156 			}
157 			denied.add(state);
158 		    }
159 		} catch (MalformedPathException e) {
160 		    // should actually never happen
161 		    log.error("internal error: malformed path exception", e);
162 		} catch (NoPrefixDeclaredException e) {
163 		    // should actually never happen
164 		    log.error("internal error: no prefix declared", e);
165 		}
166 	    }
167 	}
168 	if (denied != null) {
169 	    accessDenied.put(events, denied);
170 	}
171     }

173     /***
174      * Dispatches the events to the <code>EventListener</code>.
175      *
176      * @param events a collection of {@link EventState}s
177      *               to dispatch.
178      */
179     void consumeEvents(EventStateCollection events) throws RepositoryException {
180 	Set denied = (Set)accessDenied.remove(events);
181 	// check permissions
182 	for (Iterator it = events.iterator(); it.hasNext(); ) {
183 	    EventState state = (EventState)it.next();
184 	    if (state.getType() == EventType.CHILD_NODE_ADDED
185 		    || state.getType() == EventType.PROPERTY_ADDED
186 		    || state.getType() == EventType.PROPERTY_CHANGED) {
187 		try {
188 		    // FIXME is there a better way to check access right
189 		    // without transforming path
190 		    Path p = Path.create(state.getParentPath(), state.getChildItemQName(), false);
191 		    if (!session.getWorkspace().getAccessManager().isGranted(
192 			    p.toJCRPath(session.getNamespaceResolver()),
193 			    Permission.READ_ITEM)) {
194 			if (denied == null) {
195 			    denied = new HashSet();
196 			}
197 			denied.add(state);
198 		    }
199 		} catch (MalformedPathException e) {
200 		    // should actually never happen
201 		    log.error("internal error: malformed path exception", e);
202 		} catch (NoPrefixDeclaredException e) {
203 		    // should actually never happen
204 		    log.error("internal error: no prefix declared", e);
205 		}
206 	    }
207 	}
208 	// check if filtered iterator has at least one event
209 	EventIterator it = new FilteredEventIterator(events, filter, denied);
210 	if (it.hasNext()) {
211 	    listener.onEvent(it);
212 	} else {
213 	    // otherwise skip this listener
214 	}
215     }

217     /***
218      * Returns <code>true</code> if this <code>EventConsumer</code> is equal to
219      * some other object, <code>false</code> otherwise.
220      * <p/>
221      * Two <code>EventConsumer</code>s are considered equal if they refer to the
222      * same <code>Session</code> and the <code>EventListener</code>s they
223      * reference are equal. Note that the <code>EventFilter</code> is ignored in
224      * this check.
225      *
226      * @param obj the reference object with which to compare.
227      * @return <code>true</code> if this <code>EventConsumer</code> is equal the
228      *         other <code>EventConsumer</code>.
229      */
230     public boolean equals(Object obj) {
231 	if (this == obj) {
232 	    return true;
233 	}
234 	if (obj instanceof EventConsumer) {
235 	    EventConsumer other = (EventConsumer) obj;
236 	    return session.equals(other.session)
237 		    && listener.equals(other.listener);
238 	}
239 	return false;
240     }

247     public int hashCode() {
248 	if (hashCode == 0) {
