35      /***
36       * Pattern used to validate and parse path elements:<p>
37       * <ul>
38       * <li>group 1 is .
39       * <li>group 2 is ..
40       * <li>group 3 is namespace prefix incl. delimiter (colon)
41       * <li>group 4 is namespace prefix excl. delimiter (colon)
42       * <li>group 5 is localName
43       * <li>group 6 is index incl. brackets
44       * <li>group 7 is index excl. brackets
45       * </ul>
46       */
47      private static final Pattern PATH_ELEMENT_PATTERN = Pattern.compile("(//.)|(//.//.)|(([^ /://[//]*'\"|](?:[^/://[//]*'\"|]*[^ /://[//]*'\"|])?):)?([^ /://[//]*'\"|](?:[^/://[//]*'\"|]*[^ /://[//]*'\"|])?)(//[([1-9]//d*)//])?");

96      public static Path create(Path master, String relJCRPath, NamespaceResolver resolver, boolean canonicalize)
97  	    throws MalformedPathException {
98  	if (relJCRPath.startsWith("/")) {
99  	    throw new MalformedPathException("'" + relJCRPath + "' is not a relative path");
100 	}
101 
102 	PathElement[] elements = parse(relJCRPath, master.getElements(), resolver);
103 	if (canonicalize) {
104 	    return new Path(elements).getCanonicalPath();
105 	} else {
106 	    return new Path(elements);
107 	}

112      * @param relPath
113      * @param canonicalize
114      * @return
115      * @throws MalformedPathException
116      */
117     public static Path create(Path master, Path relPath, boolean canonicalize)
118 	    throws MalformedPathException {
119 	if (relPath.isAbsolute()) {
120 	    throw new MalformedPathException("relPath is not a relative path");
121 	}
122 
123 	PathBuilder pb = new PathBuilder(master.getElements());
124 	pb.addAll(relPath.getElements());
125 
126 	if (canonicalize) {
127 	    return pb.getPath().getCanonicalPath();
128 	} else {
129 	    return pb.getPath();
130 	}
131     }

173     /***
174      * Checks if <code>jcrPath</code> is a valid JCR-style absolute or relative
175      * path.
176      *
177      * @param jcrPath the path to be checked
178      * @throws MalformedPathException If <code>jcrPath</code> is not a valid
179      *                                JCR-style path.
180      */
181     public static void checkFormat(String jcrPath) throws MalformedPathException {
182 	if (jcrPath == null || jcrPath.length() == 0) {
183 	    throw new MalformedPathException("empty path");
184 	}
185 	// shortcut
186 	if (jcrPath.equals("/")) {
187 	    return;
188 	}
189 
190 	// split path into path elements
191 	String[] elems = jcrPath.split("/", -1);
192 	for (int i = jcrPath.startsWith("/") ? 1 : 0; i < elems.length; i++) {
193 	    // validate path element
194 	    String elem = elems[i];
195 	    Matcher matcher = PATH_ELEMENT_PATTERN.matcher(elem);
196 	    if (!matcher.matches()) {
197 		// illegal syntax for path element
198 		throw new MalformedPathException("'" + jcrPath + "' is not a valid path: '" + elem + "' is not a legal path element");
199 	    }
200 	}
201     }

204     /***
205      * Tests whether this path represents the root path, i.e. "/".
206      *
207      * @return true if this path represents the root path; false otherwise.
208      */
209     public boolean denotesRoot() {
210 	return elements.length == 1 && elements[0].denotesRoot();
211     }

213     /***
214      * Tests whether this path is absolute, i.e. whether it starts with "/".
215      *
216      * @return true if this path is absolute; false otherwise.
217      */
218     public boolean isAbsolute() {
219 	return elements.length > 0 && elements[0].denotesRoot();
220     }

222     /***
223      * Tests whether this path is canonical, i.e. whether it is absolute and
224      * does not contain redundant names such as "." and "..".
225      *
226      * @return true if this path is canonical; false otherwise.
227      * @see #isAbsolute()
228      */
229     public boolean isCanonical() {
230 	if (!isAbsolute()) {
231 	    return false;
232 	}
233 	// check path for "." and ".." names
234 	for (int i = 0; i < elements.length; i++) {
235 	    if (elements[i].equals(CURRENT_ELEMENT)
236 		    || elements[i].equals(PARENT_ELEMENT)) {
237 		return false;
238 	    }
239 	}
240 	return true;
241     }
242 
243     /***
244      * Returns the canonical path representation of this path. This typically
245      * involves removing/resolving redundant names such as "." and ".." from
246      * the path.
247      *
248      * @return a canonical path representation of this path
249      * @throws MalformedPathException if this path can not be canonicalized
250      *                                (e.g. if it is relative)
251      */
252     public Path getCanonicalPath() throws MalformedPathException {
253 	if (isCanonical()) {
254 	    return this;
255 	}
256 	if (!isAbsolute()) {
257 	    throw new MalformedPathException("only an absolute path can be canonicalized.");
258 	}
259 
260 	LinkedList queue = new LinkedList();
261 	for (int i = 0; i < elements.length; i++) {
262 	    PathElement elem = elements[i];
263 
264 	    if (elem.equals(CURRENT_ELEMENT)) {
265 		continue;
266 	    } else if (elem.equals(PARENT_ELEMENT)) {
267 		if (queue.size() <= 1) {
268 		    // the first element is the root element;
269 		    // ".." would refer to the parent of root
270 		    throw new MalformedPathException("path can not be canonicalized: unresolvable '..' element");
271 		}
272 		queue.removeLast();
273 	    } else {
274 		queue.add(elem);
275 	    }
276 	}
277 	return new Path((PathElement[]) queue.toArray(new PathElement[queue.size()]));
278     }

280     /***
281      * Returns the ancestor path of the specified relative degree.
282      * <p/>
283      * An ancestor of relative degree <i>x</i> is the path that is <i>x</i>
284      * levels up along the path.
285      * <ul>
286      * <li><i>degree</i> = 0 returns this path.
287      * <li><i>degree</i> = 1 returns the parent of this path.
288      * <li><i>degree</i> = 2 returns the grandparent of this path.
289      * <li>And so on to <i>degree</i> = <i>n</i>, where <i>n</i> is the depth
290      * of this path, which returns the root path.
291      * </ul>
292      *
293      * @param degree the relative degree of the requested ancestor.
294      * @return the ancestor path of the specified degree.
295      * @throws PathNotFoundException    if there is no ancestor of the specified
296      *                                  degree
297      * @throws IllegalArgumentException if <code>degree</code> is negative
298      */
299     public Path getAncestor(int degree) throws PathNotFoundException {
300 	if (degree < 0) {
301 	    throw new IllegalArgumentException("degree must be >= 0");
302 	} else if (degree == 0) {
303 	    return this;
304 	}
305 	int length = elements.length - degree;
306 	if (length < 1) {
307 	    throw new PathNotFoundException("no such ancestor path of degree " + degree);
308 	}
309 	PathElement[] elements = new PathElement[length];
310 	for (int i = 0; i < length; i++) {
311 	    elements[i] = this.elements[i];
312 	}
313 	return new Path(elements);
314     }

317      * Returns the number of ancestors of this path.
318      *
319      * @return the number of ancestors of this path
320      */
321     public int getAncestorCount() {
322 	return elements.length - 1;
323     }

325     /***
326      * Determines if <i>this</i> path is an ancestor of the specified path.
327      *
328      * @return <code>true</code> if <code>other</code> is a descendant;
329      *         otherwise <code>false</code>
330      * @throws MalformedPathException if either the specified path or this path
331      *                                is a relative path.
332      */
333     public boolean isAncestorOf(Path other) throws MalformedPathException {
334 	if (equals(other)) {
335 	    return false;
336 	}
337 	if (other == null) {
338 	    throw new IllegalArgumentException("null argument");
339 	}
340 	// make sure we're comparing canonical paths
341 	Path p0 = getCanonicalPath();
342 	Path p1 = other.getCanonicalPath();
343 	if (p0.elements.length >= p1.elements.length) {
344 	    return false;
345 	}
346 	for (int i = 0; i < p0.elements.length; i++) {
347 	    if (!p0.elements[i].equals(p1.elements[i])) {
348 		 return false;
349 	    }
350 	}
351 	return true;
352     }

354     /***
355      * Determines if <i>this</i> path is a descendant of the specified path.
356      *
357      * @return <code>true</code> if <code>other</code> is an ancestor;
358      *         otherwise <code>false</code>
359      * @throws MalformedPathException if either the specified path or this path
360      *                                is a relative path.
361      */
362     public boolean isDescendantOf(Path other) throws MalformedPathException {
363 	if (equals(other)) {
364 	    return false;
365 	}
366 	if (other == null) {
367 	    throw new IllegalArgumentException("null argument");
368 	}
369 	return other.isAncestorOf(this);
370     }

390     /***
391      * @param resolver
392      * @return
393      * @throws NoPrefixDeclaredException
394      */
395     public String toJCRPath(NamespaceResolver resolver) throws NoPrefixDeclaredException {
396 	if (denotesRoot()) {
397 	    // shortcut
398 	    return "/";
399 	}
400 	StringBuffer sb = new StringBuffer();
401 	for (int i = 0; i < elements.length; i++) {
402 	    if (i > 0) {
403 		sb.append('/');
404 	    }
405 	    PathElement element = elements[i];
406 	    // name
407 	    sb.append(element.toJCRName(resolver));
408 	}
409 	return sb.toString();
410     }

412     /***
413      * Returns the internal string representation of this <code>Path</code>.
414      * <p/>
415      * Note that the returned string is not a valid JCR path, i.e. the
416      * namespace URI's of the individual path elements are not replaced with
417      * their mapped prefixes. Call
418      * <code>{@link #toJCRPath(NamespaceResolver)}</code>
419      * for a JCR path representation.
420      *
421      * @return the internal string representation of this <code>Path</code>.
422      */
423     public String toString() {
424 	// Path is immutable, we can store the string representation
425 	if (string == null) {
426 	    StringBuffer sb = new StringBuffer();
427 	    for (int i = 0; i < elements.length; i++) {
428 		if (i > 0) {
429 		    // @todo find safe path separator char that does not conflict with chars in serialized QName
430 		    sb.append('\t');
431 		}
432 		PathElement element = elements[i];
433 		String elem = element.toString();
434 		sb.append(elem);
435 	    }
436 	    string = sb.toString();
437 	}
438 	return string;
439     }

441     /***
442      * Returns a <code>Path</code> holding the value of the specified
443      * string. The string must be in the format returned by the
444      * <code>Path.toString()</code> method.
445      *
446      * @param s a <code>String</code> containing the <code>Path</code>
447      *          representation to be parsed.
448      * @return the <code>Path</code> represented by the argument
449      * @throws IllegalArgumentException if the specified string can not be parsed
450      *                                  as a <code>Path</code>.
451      * @see #toString()
452      */
453     public static Path valueOf(String s) {
454 	if ("".equals(s) || s == null) {
455 	    throw new IllegalArgumentException("invalid Path literal");
456 	}
457 
458 	// split into path elements
459 
460 	// @todo find safe path separator char that does not conflict with chars in serialized QName
461 	String[] elements = s.split("\t", -1);
462 	ArrayList list = new ArrayList();
463 	for (int i = 0; i < elements.length; i++) {
464 	    String elem = elements[i];
465 	    list.add(PathElement.fromString(elem));
466 	}
467 
468 	return new Path((PathElement[]) list.toArray(new PathElement[list.size()]));
469     }

471     /***
472      * Returns a hash code value for this path.
473      *
474      * @return a hash code value for this path.
475      * @see java.lang.Object#hashCode()
476      */
477     public int hashCode() {
478 	if (hashCode == 0) {
479 	    hashCode = 1;
480 	    for (int i = 0; i < elements.length; i++) {
481 		hashCode = 43 * hashCode + elements[i].hashCode();
482 	    }
483 	}
484 	return hashCode;
485     }

488      * Compares the specified object with this path for equality.
489      *
490      * @param obj the object to be compared for equality with this path.
491      * @return <tt>true</tt> if the specified object is equal to this path.
492      */
493     public boolean equals(Object obj) {
494 	if (this == obj) {
495 	    return true;
496 	}
497 	if (obj instanceof Path) {
498 	    Path other = (Path) obj;
499 	    return Arrays.equals(elements, other.elements);
500 	}
501 	return false;
502     }

505     public static final class PathBuilder implements Cloneable {
506 	private final LinkedList queue;
507 
508 	public PathBuilder() {
509 	    queue = new LinkedList();
510 	}
511 
512 	public PathBuilder(PathElement[] elements) {
513 	    this();
514 	    addAll(elements);
515 	}
516 
517 	public void addRoot() {
518 	    queue.addFirst(ROOT_ELEMENT);
519 	}
520 
521 	public void addAll(PathElement[] elements) {
522 	    for (int i = 0; i < elements.length; i++) {
523 		queue.add(elements[i]);
524 	    }
525 	}
526 
527 	public void addFirst(String nameSpaceURI, String localName) {
528 	    queue.addFirst(new PathElement(nameSpaceURI, localName));
529 	}
530 
531 	public void addFirst(String nameSpaceURI, String localName, int index) {
532 	    queue.addFirst(new PathElement(nameSpaceURI, localName, index));
533 	}
534 
535 	public void addFirst(QName name) {
536 	    queue.addFirst(new PathElement(name));
537 	}
538 
539 	public void addFirst(QName name, int index) {
540 	    queue.addFirst(new PathElement(name, index));
541 	}
542 
543 	public void addLast(String nameSpaceURI, String localName) {
544 	    queue.addLast(new PathElement(nameSpaceURI, localName));
545 	}
546 
547 	public void addLast(String nameSpaceURI, String localName, int index) {
548 	    queue.addLast(new PathElement(nameSpaceURI, localName, index));
549 	}
550 
551 	public void addLast(QName name) {
552 	    queue.addLast(new PathElement(name));
553 	}
554 
555 	public void addLast(QName name, int index) {
556 	    queue.addLast(new PathElement(name, index));
557 	}
558 
559 	public Path getPath() throws MalformedPathException {
560 	    PathElement[] elements = (PathElement[]) queue.toArray(new PathElement[queue.size()]);
561 	    // validate path
562 	    if (elements.length == 0) {
563 		throw new MalformedPathException("empty path");
564 	    }
565 	    for (int i = 1; i < elements.length; i++) {
566 		if (elements[i].denotesRoot()) {
567 		    throw new MalformedPathException("path contains invalid root element(s)");
568 		}
569 		String localName = elements[i].getName().getLocalName();
570 		Matcher matcher = PATH_ELEMENT_PATTERN.matcher(localName);
571 		if (!matcher.matches()) {
572 		    // illegal syntax for path element
573 		    throw new MalformedPathException(localName + "' is not a legal path element");
574 		}
575 	    }
576 	    return new Path(elements);
577 	}
578 
579 	public Object clone() {
580 	    PathBuilder clone = new PathBuilder();
581 	    clone.queue.addAll(queue);
582 	    return clone;
583 	}
584     }

646     public static class PathElement {
647 
648 	private final QName name;
649 
650 	/***
651 	 * 1-based index; 0 if not explicitly specified (which is equivalent to
652 	 * specifying 1)
653 	 */
654 	private final int index;
655 
656 	private PathElement(String namespaceURI, String localName) {
657 	    this(new QName(namespaceURI, localName));
658 	}
659 
660 	private PathElement(String namespaceURI, String localName, int index) {
661 	    this(new QName(namespaceURI, localName), index);
662 	}
663 
664 	private PathElement(QName name) {
665 	    if (name == null) {
666 		throw new IllegalArgumentException("name must not be null");
667 	    }
668 	    this.name = name;
669 	    this.index = 0;
670 	}
671 
672 	private PathElement(QName name, int index) {
673 	    if (name == null) {
674 		throw new IllegalArgumentException("name must not be null");
675 	    }
676 	    if (index < 1) {
677 		throw new IllegalArgumentException("index is 1-based");
678 	    }
679 	    this.index = index;
680 	    this.name = name;
681 	}
682 
683 	/***
684 	 * Returns the name of this path element.
685 	 *
686 	 * @return the name
687 	 */
688 	public QName getName() {
689 	    return name;
690 	}
691 
692 	/***
693 	 * Returns the 1-based index or 0 if no index was specified (which is
694 	 * equivalent to specifying 1).
695 	 *
696 	 * @return Returns the 1-based index or 0 if no index was specified.
697 	 */
698 	public int getIndex() {
699 	    return index;
700 	}
701 
702 	public boolean denotesRoot() {
703 	    return false;
704 	}
705 
706 	public String toJCRName(NamespaceResolver resolver) throws NoPrefixDeclaredException {
707 	    StringBuffer sb = new StringBuffer();
708 	    // name
709 	    sb.append(name.toJCRName(resolver));
710 	    // index
711 	    int index = getIndex();
712 	    /*** FIXME the [1] subscript should only be suppressed if the item
713 	     * in question can't have same-name siblings.
714 	     */
715 	    //if (index > 0) {
716 	    if (index > 1) {
717 		sb.append('[');
718 		sb.append(index);
719 		sb.append(']');
720 	    }
721 	    return sb.toString();
722 	}
723 
724 	public String toString() {
725 	    StringBuffer sb = new StringBuffer();
726 	    // name
727 	    sb.append(name.toString());
728 	    // index
729 	    int index = getIndex();
730 	    if (index > 0) {
731 		sb.append('[');
732 		sb.append(index);
733 		sb.append(']');
734 	    }
735 	    return sb.toString();

791     private static PathElement[] parse(String jcrPath, PathElement[] master, NamespaceResolver resolver)
792 	    throws MalformedPathException {
793 	// shortcut
794 	if (jcrPath.equals("/")) {
795 	    return new PathElement[]{ROOT_ELEMENT};
796 	}
797 
798 	// split path into path elements
799 	String[] elems = jcrPath.split("/", -1);
800 	if (elems.length == 0) {
801 	    throw new MalformedPathException("empty path");
802 	}
803 
804 	ArrayList list = new ArrayList();
805 	if (master != null) {
806 	    // a master path was specified; the 'path' argument is assumed
807 	    // to be a relative path
808 	    for (int i = 0; i < master.length; i++) {
809 		list.add(master[i]);
810 	    }
811 	}
812 
813 	for (int i = 0; i < elems.length; i++) {
814 	    // validate & parse path element
815 	    String prefix;
816 	    String localName;
817 	    int index;
818 
819 	    String elem = elems[i];
820 	    if (i == 0 && elem.length() == 0) {
821 		// path is absolute, i.e. the first element is the root element
822 		if (!list.isEmpty()) {
823 		    throw new MalformedPathException("'" + jcrPath + "' is not a relative path");
824 		}
825 		list.add(ROOT_ELEMENT);
826 		continue;
827 	    }
828 	    Matcher matcher = PATH_ELEMENT_PATTERN.matcher(elem);
829 	    if (matcher.matches()) {
830 		if (matcher.group(1) != null) {
831 		    // group 1 is .
832 		    list.add(CURRENT_ELEMENT);
833 		} else if (matcher.group(2) != null) {
834 		    // group 2 is ..
835 		    list.add(PARENT_ELEMENT);
836 		} else {
837 		    // element is a name
838 
839 		    // check for prefix (group 3)
840 		    if (matcher.group(3) != null) {
841 			// prefix specified
842 			// group 4 is namespace prefix excl. delimiter (colon)
843 			prefix = matcher.group(4);
844 		    } else {
845 			// no prefix specified
846 			prefix = "";
847 		    }
848 
849 		    // group 5 is localName
850 		    localName = matcher.group(5);
851 
852 		    // check for index (group 6)
853 		    if (matcher.group(6) != null) {
854 			// index specified
855 			// group 7 is index excl. brackets
856 			index = Integer.parseInt(matcher.group(7));
857 		    } else {
858 			// no index specified
859 			index = 0;
860 		    }
861 
862 		    String nsURI;
863 		    try {
864 			nsURI = resolver.getURI(prefix);
865 		    } catch (NamespaceException nse) {
866 			// unknown prefix
867 			throw new MalformedPathException("'" + jcrPath + "' is not a valid path: '" + elem + "' specifies an unmapped namespace prefix");
868 		    }
869 
870 		    PathElement element;
871 		    if (index == 0) {
872 			element = new PathElement(nsURI, localName);
873 		    } else {
874 			element = new PathElement(nsURI, localName, index);
875 		    }
876 		    list.add(element);
877 		}
878 	    } else {
879 		// illegal syntax for path element
880 		throw new MalformedPathException("'" + jcrPath + "' is not a valid path: '" + elem + "' is not a legal path element");
881 	    }
882 	}
883 	return (PathElement[]) list.toArray(new PathElement[list.size()]);
884     }
