40      /***
41       * flags defining the current status of this <code>ItemState</code> instance

44      public static final int STATUS_UNDEFINED = 0;
45      /***
46       * 'existing', i.e. persistent state
47       */
48      public static final int STATUS_EXISTING = 1;
49      /***
50       * 'existing', i.e. persistent state that has been transiently modified (copy-on-write)
51       */
52      public static final int STATUS_EXISTING_MODIFIED = 2;
53      /***
54       * 'existing', i.e. persistent state that has been transiently removed (copy-on-write)
55       */
56      public static final int STATUS_EXISTING_REMOVED = 3;
57      /***
58       * 'new' state
59       */
60      public static final int STATUS_NEW = 4;
61      /***
62       * 'existing', i.e. persistent state that has been persistently modified by somebody else
63       */
64      public static final int STATUS_STALE_MODIFIED = 5;
65      /***
66       * 'existing', i.e. persistent state that has been destroyed by somebody else
67       */
68      public static final int STATUS_STALE_DESTROYED = 6;

70      protected int status = STATUS_UNDEFINED;

72      /***
73       * the uuid of the (primary) parent node or <code>null</code> if this is the root node
74       */

84       * Listeners (soft references)

86      protected final transient Map listeners = Collections.synchronizedMap(new ReferenceMap(ReferenceMap.SOFT, ReferenceMap.SOFT));

88      // the backing persistent item state (may be null)
89      private transient ItemState overlayedState;

91      /***
92       * Protected constructor
93       *
94       * @param parentUUID    the UUID of the (primary) parent node or <code>null</code>
95       * @param id            the id of the item state object
96       * @param initialStatus the initial status of the item state object
97       */
98      protected ItemState(String parentUUID, ItemId id, int initialStatus) {
99  	switch (initialStatus) {
100 	    case STATUS_EXISTING:
101 	    case STATUS_NEW:
102 		status = initialStatus;
103 		break;
104 	    default:
105 		String msg = "illegal status: " + initialStatus;
106 		log.error(msg);
107 		throw new IllegalArgumentException(msg);
108 	}
109 	this.id = id;
110 	this.parentUUID = parentUUID;
111 	baseVersionID = "v0.0";
112 	// @todo use modification count instead of ms (not precise enough)
113 	lastModified = System.currentTimeMillis();
114 	overlayedState = null;
115     }
116 
117     /***
118      * Protected constructor
119      *
120      * @param overlayedState backing persistent item state
121      * @param initialStatus  the initial status of the new <code>ItemState</code> instance
122      */
123     protected ItemState(ItemState overlayedState, int initialStatus) {
124 	switch (initialStatus) {
125 	    case STATUS_EXISTING_MODIFIED:
126 	    case STATUS_EXISTING_REMOVED:
127 		status = initialStatus;
128 		break;
129 	    default:
130 		String msg = "illegal status: " + initialStatus;
131 		log.error(msg);
132 		throw new IllegalArgumentException(msg);
133 	}
134 	parentUUID = overlayedState.parentUUID;
135 	baseVersionID = overlayedState.getBaseVersionID();
136 	lastModified = overlayedState.getLastModified();
137 	id = overlayedState.getId();
138 	this.overlayedState = overlayedState;
139 	// add this transient state as a listener on the overlayed state
140 	this.overlayedState.addListener(this);
141     }

143     /***
144      * Called by <code>TransientItemStateManager</code> when this item state
145      * is disposed.
146      */
147     void onDisposed() {
148 	// prepare this instance so it can be gc'ed
149 	listeners.clear();
150 	if (overlayedState != null) {
151 	    overlayedState.removeListener(this);
152 	    overlayedState = null;
153 	}
154 	status = STATUS_UNDEFINED;
155     }

157     /***
158      * Notify the listeners that the persistent state this object is
159      * representing has been discarded.
160      */
161     protected void notifyStateDiscarded() {
162 	// copy listeners to array to avoid ConcurrentModificationException
163 	ItemStateListener[] la = new ItemStateListener[listeners.size()];
164 	Iterator iter = listeners.values().iterator();
165 	int cnt = 0;
166 	while (iter.hasNext()) {
167 	    la[cnt++] = (ItemStateListener) iter.next();
168 	}
169 	for (int i = 0; i < la.length; i++) {
170 	    if (la[i] != null) {
171 		la[i].stateDiscarded(this);
172 	    }
173 	}
174     }

294     /***
295      * Sets the new status of this item.
296      *
297      * @param newStatus the new status
298      */
299     public void setStatus(int newStatus) {
300 	switch (newStatus) {
301 	    case ItemState.STATUS_NEW:
302 	    case ItemState.STATUS_EXISTING:
303 	    case ItemState.STATUS_EXISTING_REMOVED:
304 	    case ItemState.STATUS_EXISTING_MODIFIED:
305 	    case ItemState.STATUS_STALE_MODIFIED:
306 	    case ItemState.STATUS_STALE_DESTROYED:
307 	    case ItemState.STATUS_UNDEFINED:
308 		status = newStatus;
309 		return;
310 	}
311 	String msg = "illegal status: " + newStatus;
312 	log.error(msg);
313 	throw new IllegalArgumentException(msg);
314     }

316     /***
317      * Discards this instance, i.e. renders it 'invalid'.
318      */
319     public void discard() {
320 	if (status != STATUS_UNDEFINED) {
321 	    // notify listeners
322 	    notifyStateDiscarded();
323 	    // reset status
324 	    status = STATUS_UNDEFINED;
325 	}
326     }
