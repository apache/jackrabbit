18  import org.apache.jackrabbit.core.QName;
19  
20  import java.io.IOException;
21  import java.io.ObjectInputStream;
22  import java.io.ObjectOutputStream;
23  
24  /***
25   * <code>PersistentPropertyState</code> represents the persistent state of a
26   * <code>Property</code>.
27   *
28   * @author Stefan Guggisberg
29   * @version $Revision: 1.9 $, $Date: 2004/08/02 16:19:48 $
30   */
31  public abstract class PersistentPropertyState extends PropertyState implements PersistableItemState {
32  
33      static final long serialVersionUID = -8919019313955817383L;
34  
35      protected final transient PersistenceManager persistMgr;
36  
37      /***
38       * Package private constructor
39       *
40       * @param name       name of the property
41       * @param parentUUID the uuid of the parent node
42       * @param persistMgr the persistence manager
43       */
44      protected PersistentPropertyState(QName name, String parentUUID, PersistenceManager persistMgr) {
45  	super(name, parentUUID, STATUS_NEW);
46  	this.persistMgr = persistMgr;
47      }
48  
49      //-------------------------------------------------< PersistableItemState >
50      /***
51       * @see PersistableItemState#reload
52       */
53      public abstract void reload() throws ItemStateException;
54  
55      /***
56       * @see PersistableItemState#store
57       */
58      public abstract void store() throws ItemStateException;
59  
60      /***
61       * @see PersistableItemState#destroy
62       */
63      public abstract void destroy() throws ItemStateException;
64  
65      //-------------------------------------------------< Serializable support >
66      private void writeObject(ObjectOutputStream out) throws IOException {
67  	// delegate to default implementation
68  	out.defaultWriteObject();
69      }
70  
71      private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
72  	// delegate to default implementation
73  	in.defaultReadObject();
74      }
75  }
