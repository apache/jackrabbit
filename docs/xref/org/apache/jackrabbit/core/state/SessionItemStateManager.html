55      private synchronized void collectDescendantItemStates(ItemId id, List descendents) {
56  	// XXX very inefficient implementation, especially if # of transient states
57  	// is relatively small compared to the total # of persistent states
58  	if (descendents.size() == transientStateMgr.getEntriesCount()) {
59  	    return;
60  	}
61  	try {
62  	    if (id.denotesNode()) {
63  		NodeId parentId = (NodeId) id;
64  		ItemId[] childIds = hierMgr.listChildren(parentId);
65  		for (int i = 0; i < childIds.length; i++) {
66  		    ItemId childId = childIds[i];
67  		    if (transientStateMgr.hasItemState(childId)) {
68  			ItemState state = transientStateMgr.getItemState(childId);
69  			if (!descendents.contains(state)) {
70  			    descendents.add(state);
71  			}
72  		    }
73  		    if (childId.denotesNode()) {
74  			// recurse
75  			collectDescendantItemStates(childId, descendents);
76  		    }
77  		}
78  		// also add transient child nodes that have been unlinked from
79  		// the specified parent node but are not orphaned yet (i.e.
80  		// they are still linked to at least one other parent node)
81  		if (transientStateMgr.hasItemState(parentId)) {
82  		    NodeState parentState = (NodeState) transientStateMgr.getItemState(parentId);
83  		    Iterator iter = parentState.getRemovedChildNodeEntries().iterator();
84  		    while (iter.hasNext()) {
85  			// removed child nodes
86  			NodeState.ChildNodeEntry cne = (NodeState.ChildNodeEntry) iter.next();
87  			NodeId removedChildId = new NodeId(cne.getUUID());
88  			if (transientStateMgr.hasItemState(removedChildId)) {
89  			    ItemState state = transientStateMgr.getItemState(removedChildId);
90  			    if (!descendents.contains(state)) {
91  				descendents.add(state);
92  			    }
93  			}
94  		    }
95  		}
96  	    }
97  	} catch (ItemStateException ise) {
98  	    log.warn("inconsistent hierarchy state", ise);
99  	} catch (RepositoryException re) {
100 	    log.warn("inconsistent hierarchy state", re);
101 	}
102     }

104     private synchronized void collectDescendantItemStatesInAttic(ItemId id, List descendents) {
105 	// XXX very inefficient implementation, especially if # of transient states
106 	// is relatively small compared to the total # of persistent states
107 	if (descendents.size() == transientStateMgr.getEntriesInAtticCount()) {
108 	    return;
109 	}
110 	try {
111 	    if (id.denotesNode()) {
112 		NodeId parentId = (NodeId) id;
113 
114 		// traverse zombie children (i.e. children marked as removed)
115 		ItemId[] childIds = hierMgr.listZombieChildren(parentId);
116 		for (int i = 0; i < childIds.length; i++) {
117 		    ItemId childId = childIds[i];
118 		    // check attic
119 		    if (transientStateMgr.hasItemStateInAttic(childId)) {
120 			// found on attic, add to descendents list
121 			ItemState state = transientStateMgr.getItemStateInAttic(childId);
122 			if (!descendents.contains(state)) {
123 			    descendents.add(state);
124 			}
125 		    }
126 		    if (childId.denotesNode()) {
127 			// recurse
128 			collectDescendantItemStatesInAttic(childId, descendents);
129 		    }
130 		}
131 
132 		// traverse existing children (because they might have zombie children)
133 		childIds = hierMgr.listChildren(parentId);
134 		for (int i = 0; i < childIds.length; i++) {
135 		    ItemId childId = childIds[i];
136 		    if (childId.denotesNode()) {
137 			// recurse
138 			collectDescendantItemStatesInAttic(childId, descendents);
139 		    }
140 		}
141 	    }
142 	} catch (ItemStateException ise) {
143 	    log.warn("inconsistent hierarchy state", ise);
144 	} catch (RepositoryException re) {
145 	    log.warn("inconsistent hierarchy state", re);
146 	}
147     }

175      * @see ItemStateProvider#getItemState(ItemId)
176      */
177     public ItemState getItemState(ItemId id)
178 	    throws NoSuchItemStateException, ItemStateException {
179 	// first check if the specified item has been transiently removed
180 	if (transientStateMgr.hasItemStateInAttic(id)) {
181 	    /***
182 	     * check if there's new transient state for the specified item
183 	     * (e.g. if a property with name 'x' has been removed and a new
184 	     * property with same name has been created);
185 	     * this will throw a NoSuchItemStateException if there's no new
186 	     * transient state
187 	     */
188 	    return transientStateMgr.getItemState(id);
189 	}
190 	try {
191 	    // check if there's transient state for the specified item
192 	    return transientStateMgr.getItemState(id);
193 	} catch (NoSuchItemStateException nsise) {
194 	    // check if there's persistent state for the specified item
195 	    return persistentStateMgr.getItemState(id);
196 	}
197     }

250     /***
251      * Returns an iterator over those transient item state instances that are
252      * direct or indirect descendents of the item state with the given
253      * <code>parentId</code>. The transient item state instance with the given
254      * <code>parentId</code> itself (if there is such) will not be included.
255      * <p/>
256      * The instances are returned in depth-first tree traversal order.
257      *
258      * @param parentId the id of the common parent of the transient item state
259      *                 instances to be returned.
260      * @return an iterator over descendant transient item state instances
261      */
262     public Iterator getDescendantTransientItemStates(ItemId parentId) {
263 	// @todo need a more efficient way to find descendents in cache (e.g. using hierarchical index)
264 	if (!transientStateMgr.hasAnyItemStates()) {
265 	    return Collections.EMPTY_LIST.iterator();
266 	}
267 	// collection of descendant transient states:
268 	// the path serves as key and sort criteria
269 	TreeMap descendants = new TreeMap(new PathComparator());
270 	try {
271 	    Path[] parentPaths = hierMgr.getAllPaths(parentId);
272 	    /***
273 	     * walk through list of transient states and check if
274 	     * they are descendants of the specified parent
275 	     */
276 	    Iterator iter = transientStateMgr.getEntries();
277 	    while (iter.hasNext()) {
278 		ItemState state = (ItemState) iter.next();
279 		ItemId id = state.getId();
280 		Path[] paths = hierMgr.getAllPaths(id);
281 		boolean isDescendant = false;
282 		/***
283 		 * check if any of the paths to the transient state
284 		 * is a descendant of any of the specified parentId's paths
285 		 */
286 		for (int i = 0; i < paths.length; i++) {
287 		    Path p0 = paths[i];	// path to transient state
288 		    // walk through array of the specified parentId's paths
289 		    for (int j = 0; j < parentPaths.length; j++) {
290 			Path p1 = parentPaths[j]; // path to specified parentId
291 			if (p0.isDescendantOf(p1)) {
292 			    // this is a descendant, add it to the list and
293 			    // continue with next transient state
294 			    descendants.put(p0, state);
295                             isDescendant = true;
296 			    break;
297 			}
298 		    }
299 		    if (isDescendant) {
300 			break;
301 		    }
302 		}
303 		if (!isDescendant && id.denotesNode()) {
304 		    /***
305 		     * finally check if transient state has been unlinked
306 		     * from a parent node (but is not orphaned yet, i.e. is
307 		     * still linked to at least one other parent node);
308 		     * if that's the case, check if that parent is a
309 		     * descendant of/identical with any of the specified
310 		     * parentId's paths.
311 		     */
312 		    NodeState nodeState = (NodeState) state;
313 		    Iterator iterUUIDs = nodeState.getRemovedParentUUIDs().iterator();
314 		    while (iterUUIDs.hasNext()) {
315 			/***
316 			 * check if any of the paths to the removed parent
317 			 * is a descendant of/identical with any of the
318 			 * specified parentId's paths.
319 			 */
320 			Path[] pa = hierMgr.getAllPaths(new NodeId((String) iterUUIDs.next()));
321 			for (int k = 0; k < pa.length; k++) {
322 			    Path p0 = pa[k];	// path to removed parent
323 			    // walk through array of the specified parentId's paths
324 			    for (int j = 0; j < parentPaths.length; j++) {
325 				Path p1 = parentPaths[j]; // path to specified parentId
326 				if (p0.equals(p1) || p0.isDescendantOf(p1)) {
327 				    // this is a descendant, add it to the list and
328 				    // continue with next transient state
329 
330 				    // FIXME need to create dummy path in order
331 				    // to avoid conflicts
332 				    Path.PathBuilder pb = new Path.PathBuilder(p0.getElements());
333 				    pb.addFirst(NamespaceRegistryImpl.NS_DEFAULT_URI, Integer.toString(new Random().nextInt()));
334 				    descendants.put(pb.getPath(), state);
335 				    isDescendant = true;
336 				    break;
337 				}
338 			    }
339 			    if (isDescendant) {
340 				break;
341 			    }
342 			}
343 			if (isDescendant) {
344 			    break;
345 			}
346 		    }
347 		}
348 		// continue with next transient state
349 	    }
350 	} catch (MalformedPathException mpe) {
351 	    log.warn("inconsistent hierarchy state", mpe);
352 	} catch (RepositoryException re) {
353 	    log.warn("inconsistent hierarchy state", re);
354 	}
355 
356 	return descendants.values().iterator();
357 /*
358 	ArrayList descendents = new ArrayList();
359 	collectDescendantItemStates(parentId, descendents);
360 	return Collections.unmodifiableList(descendents).iterator();
361 */
362     }

551     /***
552      * Comparator used to sort canonical <code>Path</code> objects
553      * hierarchically (in depth-first tree traversal order).
554      */
555     static class PathComparator implements Comparator {
556 	/***
557 	 *
558 	 * @param o1
559 	 * @param o2
560 	 * @return
561 	 */
562 	public int compare(Object o1, Object o2) {
563 	    Path p1 = (Path) o1;
564 	    Path p2 = (Path) o2;
565 	    if (p1.equals(p2)) {
566 		return 0;
567 	    }
568 	    try {
569 		if (p1.isAncestorOf(p2)) {
570 		    return -1;
571 		} else if (p1.isDescendantOf(p2)) {
572 		    return 1;
573 		}
574 	    } catch (MalformedPathException mpe) {
575 		log.warn("unable to compare non-canonical (i.e. relative) paths", mpe);
576 	    }
577 	    // the 2 paths are not on the same graph;
578 	    // do string comparison of individual path elements
579 	    Path.PathElement[] pea1 = p1.getElements();
580 	    Path.PathElement[] pea2 = p2.getElements();
581 	    for (int i = 0; i < pea1.length; i++) {
582 		if (i >= pea2.length) {
583 		    return 1;
584 		}
585 		String s1 = pea1[i].toString();
586 		String s2 = pea2[i].toString();
587 		int result = s1.compareTo(s2);
588 		if (result != 0) {
589 		    return result;
590 		}
591 	    }
592 	    return 0;
593 	}
