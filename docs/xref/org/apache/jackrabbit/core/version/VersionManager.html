37      private static Logger log = Logger.getLogger(VersionManager.class);

39      // root path for version storage
40      public static final QName VERSION_HISTORY_ROOT_NAME =
41  	    new QName(NamespaceRegistryImpl.NS_JCR_URI, "versionStorage");
42  
43      // the system session for the versioning
44      private final SessionImpl session;
45  
46      /***
47       * the root node of the version histories
48       */
49      private final NodeImpl historyRoot;
50  
51      /***
52       * Creates a new VersionManager.
53       *
54       * @param session
55       * @throws RepositoryException
56       */
57      public VersionManager(SessionImpl session) throws RepositoryException {
58  	this.session = session;
59  
60  	// check for versionhistory root
61  	NodeImpl systemRoot = ((RepositoryImpl) session.getRepository()).getSystemRootNode(session);
62  	if (!systemRoot.hasNode(VERSION_HISTORY_ROOT_NAME)) {
63  	    // if not exist, create
64  	    systemRoot.addNode(VERSION_HISTORY_ROOT_NAME, NodeTypeRegistry.NT_UNSTRUCTURED);
65  	    systemRoot.save();
66  	}
67  	historyRoot = systemRoot.getNode(VERSION_HISTORY_ROOT_NAME);
68      }
69  
70      /***
71       * Creates a new Version History and returns the UUID of it.
72       *
73       * @param node the node for which the version history is to be initialized
74       * @return the UUID of the new version history node
75       * @throws RepositoryException
76       */
77      public VersionHistory createVersionHistory(NodeImpl node)
78  	    throws RepositoryException {
79  
80  	// check if history already exists
81  	QName historyNodeName = new QName(NamespaceRegistryImpl.NS_DEFAULT_URI, node.getUUID());
82  	if (historyRoot.hasNode(historyNodeName)) {
83  	    //throw new RepositoryException("Unable to initialize version history. Already exists");
84  	    historyRoot.getNode(historyNodeName).remove(".");
85  	}
86  
87  	// create new history node
88  	VersionHistoryImpl vh =	VersionHistoryImpl.create(historyRoot.addNode(historyNodeName,
89  		NodeTypeRegistry.NT_VERSION_HISTORY));
90  
91  	// and initialize the root version
92  	((VersionImpl) vh.getRootVersion()).initFrozenState(node);
93  
94  	// save new history
95  	historyRoot.save();
96  
97  	// must aquire version history with the node's session
98  	return VersionHistoryImpl.create((NodeImpl) node.getSession().getNodeByUUID(vh.getUUID()));
99      }

124     /***
125      * Checks in a node
126      *
127      * @param node
128      * @return
129      * @throws RepositoryException
130      * @see Node#checkin()
131      */
132     public Version checkin(NodeImpl node) throws RepositoryException {
133 	// assuming node is versionable and checkout (check in nodeimpl)
134 	// To create a new version of a versionable node N, the client calls N.checkin.
135 	// This causes the following series of events:
136 
137 	// 0. resolve the predecessors
138 	Value[] values = node.getProperty(VersionImpl.PROPNAME_PREDECESSORS).getValues();
139 	VersionImpl[] preds = new VersionImpl[values.length];
140 	for (int i = 0; i < values.length; i++) {
141 	    preds[i] = (VersionImpl) node.getSession().getNodeByUUID(values[i].getString());
142 	}
143 
144 	// 0.1 search a predecessor, suitable for generating the new name
145 	String versionName = null;
146 	int maxDots = Integer.MAX_VALUE;
147 	for (int i = 0; i < preds.length; i++) {
148 	    // take the first pred. without a successor
149 	    if (preds[i].getSuccessors().length == 0) {
150 		versionName = preds[i].getName();
151 		// need to count the dots
152 		int pos = -1;
153 		int numDots = 0;
154 		while (versionName.indexOf('.', pos + 1) >= 0) {
155 		    pos = versionName.indexOf('.', pos + 1);
156 		    numDots++;
157 		}
158 		if (numDots < maxDots) {
159 		    maxDots = numDots;
160 		    versionName = pos < 0 ? "1.0" : versionName.substring(0, pos + 1) + (Integer.parseInt(versionName.substring(pos + 1)) + 1);
161 		}
162 		break;
163 	    }
164 	}
165 	// if no empty found, generate new name
166 	VersionHistoryImpl vh = getVersionHistory(node);
167 	if (versionName == null) {
168 	    versionName = preds[0].getName();
169 	    do {
170 		versionName += ".1";
171 	    } while (vh.hasNode(versionName));
172 	}
173 
174 	try {
175 	    // 1. A new nt:version node V is created and added as a child node to VH,
176 	    //    the nt:versionHistory pointed to by N’s jcr:versionHistory property.
177 	    VersionImpl v = (VersionImpl) vh.addNode(versionName,
178 		    NodeTypeRegistry.NT_VERSION.toJCRName(session.getNamespaceResolver()));
179 
180 	    // 3. N’s base version is changed to V by altering N’s jcr:baseVersion
181 	    //    property to point to V.
182 	    //   (will be done in the nodeimpl)
183 
184 	    // 4. N’s checked-in/checked-out status is changed to checked-in by
185 	    //    changing its jcr:isCheckedOut property to false.
186 	    //    (will be done in NodeImpl)
187 
188 	    // 5. The state of N is recorded in V by storing information about
189 	    //    N’s child items (properties or child nodes) to V, as prescribed by
190 	    //    the OnParentVersion attribute of each of N’s child items.
191 	    //    See 7.2.8, below, for the details. The jcr:primaryType,
192 	    //    jcr:mixinTypes and jcr:uuid properties of N are copied over to V
193 	    //    but renamed to jcr:frozenPrimaryType, jcr:frozenMixinTypes and
194 	    //    jcr:frozenUUID to avoid conflict with V's own properties with these names.
195 	    v.createFrozenState(node);
196 
197 	    // 2. N’s current jcr:predecessors property is copied to V, and N’s
198 	    //    jcr:predecessors property is then set to null.  A reference to V
199 	    //    is then added to the jcr:successors property of each of the versions
200 	    //    identified in V’s jcr:predecessors property.
201 	    InternalValue[] ivPreds = new InternalValue[preds.length];
202 	    for (int i = 0; i < preds.length; i++) {
203 		ivPreds[i] = InternalValue.create(new UUID(preds[i].getUUID()));
204 		preds[i].internalAddSuccessor(v);
205 	    }
206 	    v.internalSetProperty(VersionImpl.PROPNAME_PREDECESSORS, ivPreds);
207 
208 	    // 6. V is given a name, sometimes based upon the name of V’s predecessor.
209 	    //    For example, an increment from “1.5” to “1.6”.
210 	    // (is done before)
211 	    vh.save();
212 	    return v;
213 	} catch (RepositoryException e) {
214 	    log.error("Aborting checkin. Error while creating version: " + e.toString());
215 	    vh.refresh(false);
216 	    throw e;
217 	} catch (NoPrefixDeclaredException npde) {
218 	    String msg = "Aborting checkin. Error while creating version: " + npde.toString();
219 	    log.error(msg, npde);
220 	    vh.refresh(false);
221 	    throw new RepositoryException(msg, npde);
222 	}
223     }
