150     /***
151      * Adds a successor to the jcr:successor list
152      * <p/>
153      * please note, that this operation might corrupt the version graph
154      *
155      * @param succ
156      */
157     void internalAddSuccessor(VersionImpl succ) throws RepositoryException {
158 	Version[] successors = getSuccessors();
159 	InternalValue[] values = new InternalValue[successors.length + 1];
160 	for (int i = 0; i < successors.length; i++) {
161 	    values[i] = InternalValue.create(new UUID(successors[i].getUUID()));
162 	}
163 	values[successors.length] = InternalValue.create(new UUID(succ.getUUID()));
164 	internalSetProperty(PROPNAME_SUCCESSORS, values);
165     }

167     /***
168      * Detaches itself from the version graph.
169      *
170      * @throws RepositoryException
171      */
172     void internalDetach() throws RepositoryException {
173 	// detach this from all successors
174 	VersionImpl[] succ = (VersionImpl[]) getSuccessors();
175 	for (int i = 0; i < succ.length; i++) {
176 	    succ[i].internalDetachPredecessor(this);
177 	}
178 	// detach this from all predecessors
179 	VersionImpl[] pred = (VersionImpl[]) getPredecessors();
180 	for (int i = 0; i < pred.length; i++) {
181 	    pred[i].internalDetachSuccessor(this);
182 	}
183 	// clear properties
184 	internalSetProperty(PROPNAME_PREDECESSORS, new InternalValue[0]);
185 	internalSetProperty(PROPNAME_SUCCESSORS, new InternalValue[0]);
186     }

188     /***
189      * Removes the successor V of this successors list and adds all of Vs
190      * successors to it.
191      * <p/>
192      * please note, that this operation might corrupt the version graph
193      *
194      * @param v the successor to detach
195      */
196     private void internalDetachSuccessor(VersionImpl v) throws RepositoryException {
197 	Version[] vsucc = v.getSuccessors();
198 	Version[] successors = getSuccessors();
199 	InternalValue[] values = new InternalValue[successors.length - 1 + vsucc.length];
200 	int idx = 0;
201 	// copy successors but ignore 'v'
202 	for (int i = 0; i < successors.length; i++) {
203 	    if (!successors[i].isSame(v)) {
204 		values[idx++] = InternalValue.create(new UUID(successors[i].getUUID()));
205 	    }
206 	}
207 	// attach v's successors
208 	for (int i = 0; i < vsucc.length; i++) {
209 	    values[idx++] = InternalValue.create(new UUID(vsucc[i].getUUID()));
210 	}
211 	internalSetProperty(PROPNAME_SUCCESSORS, values);
212     }

214     /***
215      * Removes the predecessor V of this predecessor list and adds all of Vs
216      * predecessors to it.
217      * <p/>
218      * please note, that this operation might corrupt the version graph
219      *
220      * @param v the successor to detach
221      */
222     private void internalDetachPredecessor(VersionImpl v) throws RepositoryException {
223 	Version[] vpred = v.getPredecessors();
224 	Version[] tpred = getPredecessors();
225 	InternalValue[] values = new InternalValue[tpred.length - 1 + vpred.length];
226 	int idx = 0;
227 	// copy predecessors but ignore 'v'
228 	for (int i = 0; i < tpred.length; i++) {
229 	    if (!tpred[i].isSame(v)) {
230 		values[idx++] = InternalValue.create(new UUID(tpred[i].getUUID()));
231 	    }
232 	}
233 	// attach v's predecessors
234 	for (int i = 0; i < vpred.length; i++) {
235 	    values[idx++] = InternalValue.create(new UUID(vpred[i].getUUID()));
236 	}
237 	internalSetProperty(PROPNAME_PREDECESSORS, values);
238     }

276     /***
277      * Initializes / Loads the version label cache
278      */
279     private void initLabelCache() throws RepositoryException {
280 	if (cachedVersionLabels == null) {
281 	    cachedVersionLabels = new HashSet();
282 	    if (hasProperty(PROPNAME_VERSION_LABELS)) {
283 		Value[] values = getProperty(PROPNAME_VERSION_LABELS).getValues();
284 		for (int i = 0; i < values.length; i++) {
285 		    cachedVersionLabels.add(values[i].getString());
286 		}
287 	    }
288 	}
289     }

291     /***
292      * Saves the current labels in the cache to the 'VersionLabels' property
293      * of this node.
294      *
295      * @throws RepositoryException
296      */
297     private void saveLabelCache() throws RepositoryException {
298 	InternalValue[] newValues = new InternalValue[cachedVersionLabels.size()];
299 	Iterator iter = cachedVersionLabels.iterator();
300 	for (int i = 0; i < newValues.length; i++) {
301 	    newValues[i] = InternalValue.create((String) iter.next());
302 	}
303 	internalSetProperty(PROPNAME_VERSION_LABELS, newValues);
304 	save();
305     }

318     /***
319      * Checks if this version is more recent than the given version <code>v</code>.
320      * A version is more recent if and only if it is a successor (or a successor
321      * of a successor, etc., to any degree of separation) of the compared one.
322      *
323      * @param v the version to check
324      * @return <code>true</code> if the version is more recent;
325      *         <code>false</code> otherwise.
326      * @throws RepositoryException if an error occurrs.
327      */
328     public boolean isMoreRecent(Version v) throws RepositoryException {
329 	VersionIteratorImpl iter = new VersionIteratorImpl(v);
330 	while (iter.hasNext()) {
331 	    if (iter.nextVersion().isSame(this)) {
332 		return true;
333 	    }
334 	}
335 	return false;
336     }
