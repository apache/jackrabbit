104     /***
105      * <p>Constructs a UUID from a 16 byte array.</p>
106      *
107      *  @param byteArray the 16 byte array to construct this UUID from.
108      *  @throws IllegalArgumentException argument must be 16 bytes
109      */
110     public UUID(byte[] byteArray) throws IllegalArgumentException {
111         super();
112         if (byteArray.length != UUID_BYTE_LENGTH) {
113             throw new IllegalArgumentException("UUID must be contructed using a 16 byte array.");
114         }
115         // UUID must be immutable so a copy is used.
116         System.arraycopy(byteArray, 0, rawBytes, 0, UUID_BYTE_LENGTH);
117     }

309      * <p>Returns the clock sequence value in the UUID. The clock sequence is a random assigned to a particular clock instance that
310      * generated the time in the timestamp of a time based UUID.</p>
311      *
312      * @return the clock sequence value in the UUID.
313      * @throws UnsupportedOperationException thrown if this is not a IETF variant or not a time-based UUID.

315     public int clockSequence() throws UnsupportedOperationException {
316         //if variant is not mealling leach salz throw unsupported operation exception
317         if (variant() != VARIANT_IETF_DRAFT || version() != VERSION_ONE) {
318             throw new UnsupportedOperationException(WRONG_VAR_VER_MSG);
319         }
320         if (clockSq == null) {
321             byte[] b = {((byte) (rawBytes[8] & 0x3F)), rawBytes[9]};
322             clockSq = new Short(Bytes.toShort(b));
323         }
324         return clockSq.intValue();
325     }

346      * <p>Returns the variant field of the UUID.</p>
347      *
348      * @return Returns the variant field of the UUID.
349      * @see UUID#VARIANT_NCS_COMPAT
350      * @see UUID#VARIANT_IETF_DRAFT
351      * @see UUID#VARIANT_MS
352      * @see UUID#VARIANT_FUTURE
353      */
354     public int variant() {
355         if (variant == -1) {
356             if ((rawBytes[8] & 0x80) == 0x0) {
357                 variant = VARIANT_NCS_COMPAT;
358             } else if ((rawBytes[8] & 0x40) == 0x0) {
359                 variant = VARIANT_IETF_DRAFT;
360             } else if ((rawBytes[8] & 0x20) == 0x0) {
361                 variant = VARIANT_MS;
362             } else {
363                 variant = VARIANT_FUTURE;
364             }
365         }
366         return variant;
367     }

397         //if variant is not mealling leach salz throw unsupported operation exception

399             throw new UnsupportedOperationException(WRONG_VAR_VER_MSG);
400         }
401         if (timestamp == -1) {
402             byte[] longVal = new byte[8];
403             System.arraycopy(rawBytes, TIME_HI_START_POS, longVal, TIME_HI_TS_POS, TIME_HI_BYTE_LEN);
404             System.arraycopy(rawBytes, TIME_MID_START_POS, longVal, TIME_MID_TS_POS, TIME_MID_BYTE_LEN);
405             System.arraycopy(rawBytes, TIME_LOW_START_POS, longVal, TIME_LOW_TS_POS, TIME_LOW_BYTE_LEN);
406             longVal[TIME_HI_TS_POS] &= 0x0F;
407             timestamp = Bytes.toLong(longVal);
408         }
409         return timestamp;
