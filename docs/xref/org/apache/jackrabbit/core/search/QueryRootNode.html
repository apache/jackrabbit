139     /***
140      * Returns a string representation of this query node including its sub-nodes.
141      * The returned string is formatted in JCRQL syntax.
142      *
143      * @return a string representation of this query node including its sub-nodes.
144      */
145     public String toJCRQLString() {
146 	StringBuffer sb = new StringBuffer("SELECT *");
147 	String comma = "";
148 	if (nodeTypes.size() > 0) {
149 	    sb.append(" FROM");
150 	}
151 	for (Iterator it = nodeTypes.iterator(); it.hasNext(); ) {
152 	    NodeTypeQueryNode nodeType = (NodeTypeQueryNode)it.next();
153 	    sb.append(comma);
154 	    sb.append(" ").append(nodeType.getValue());
155 	    comma = ",";
156 	}
157 	if (locationNode != null) {
158 	    sb.append(" ").append(locationNode.toJCRQLString());
159 	}
160 	if (constraintNode != null) {
161 	    sb.append(" WHERE ").append(constraintNode.toJCRQLString());
162 	}
163 	return sb.toString();
164     }
