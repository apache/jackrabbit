58      /***
59       * Returns a string representation of this query node including its sub-nodes.
60       * The returned string is formatted in JCRQL syntax.
61       *
62       * @return a string representation of this query node including its sub-nodes.
63       */
64      public String toJCRQLString() {
65  	StringBuffer sb = new StringBuffer();
66  	boolean bracket = false;
67  	if (getParent() instanceof NotQueryNode) {
68  	    bracket = true;
69  	}
70  	if (bracket) sb.append("(");
71  	String and = "";
72  	for (Iterator it = operands.iterator(); it.hasNext(); ) {
73  	    sb.append(and);
74  	    sb.append(((QueryNode)it.next()).toJCRQLString());
75  	    and = " AND ";
76  	}
77  	if (bracket) sb.append(")");
78  	return sb.toString();
79      }
