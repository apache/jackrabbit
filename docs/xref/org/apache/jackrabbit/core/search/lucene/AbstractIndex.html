16  package org.apache.jackrabbit.core.search.lucene;

18  import org.apache.lucene.store.Directory;
19  import org.apache.lucene.index.IndexWriter;
20  import org.apache.lucene.index.IndexReader;
21  import org.apache.lucene.index.Term;
22  import org.apache.lucene.analysis.Analyzer;
23  import org.apache.lucene.document.Document;
24  import org.apache.lucene.search.IndexSearcher;
25  
26  import java.io.IOException;
27  
28  /***
29   *
30   * @author Marcel Reutegger
31   * @version $Revision:  $, $Date:  $
32   */
33  abstract class AbstractIndex {
34  
35      private IndexWriter indexWriter;
36  
37      private IndexReader indexReader;
38  
39      private Directory directory;
40  
41      private Analyzer analyzer;
42  
43      private boolean useCompoundFile = false;
44  
45      AbstractIndex(Analyzer analyzer, Directory directory) throws IOException {
46  	this.analyzer = analyzer;
47  	this.directory = directory;
48  
49  	if (!IndexReader.indexExists(directory)) {
50  	    indexWriter = new IndexWriter(directory, analyzer, true);
51  	    indexWriter.setUseCompoundFile(useCompoundFile);
52  	}
53      }
54  
55      synchronized void setUseCompoundFile(boolean b) {
56  	useCompoundFile = b;
57  	if (indexWriter != null) {
58  	    indexWriter.setUseCompoundFile(b);
59  	}
60      }
61  
62  
63      /***
64       * Default implementation returns the same instance as passed
65       * in the constructor.
66       * @return the directory instance passed in the constructor
67       * @throws IOException
68       */
69      Directory getDirectory() throws IOException {
70  	return this.directory;
71      }
72  
73      IndexSearcher getIndexSearcher() throws IOException {
74  	return new IndexSearcher(getIndexReader());
75      }
76  
77      void addDocument(Document doc) throws IOException {
78  	getIndexWriter().addDocument(doc);
79      }
80  
81      void removeDocument(Term idTerm) throws IOException {
82  	getIndexReader().delete(idTerm);
83      }
84  
85      synchronized protected IndexReader getIndexReader() throws IOException {
86  	if (indexWriter != null) {
87  	    indexWriter.close();
88  	    indexWriter = null;
89  	}
90  	if (indexReader == null) {
91  	    indexReader = IndexReader.open(getDirectory());
92  	}
93  	return indexReader;
94      }
95  
96      synchronized protected IndexWriter getIndexWriter() throws IOException {
97  	if (indexReader != null) {
98  	    indexReader.close();
99  	    indexReader = null;
100 	}
101 	if (indexWriter == null) {
102 	    indexWriter = new IndexWriter(getDirectory(), analyzer, false);
103 	    indexWriter.setUseCompoundFile(useCompoundFile);
104 	}
105 	return indexWriter;
106     }
107 
108     void close() {
109 	if (indexWriter != null) {
110 	    try {
111 		indexWriter.close();
112 	    } catch (IOException e) {
113 		// FIXME do logging
114 	    }
115 	    indexWriter = null;
116 	}
117 	if (indexReader != null) {
118 	    try {
119 		indexReader.close();
120 	    } catch (IOException e) {
121 		// FIXME do logging
122 	    }
123 	    indexReader = null;
124 	}
125 	if (directory != null) {
126 	    try {
127 		directory.close();
128 	    } catch (IOException e) {
129 		directory = null;
130 	    }
131 	}
132     }
133 }
