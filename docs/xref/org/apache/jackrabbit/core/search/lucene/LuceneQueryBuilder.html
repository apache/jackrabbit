64      public static Query createQuery(QueryRootNode root,
65  				    SessionImpl session,
66  				    NamespaceMappings nsMappings,
67  				    Analyzer analyzer)
68  	    throws RepositoryException {
69  
70  	LuceneQueryBuilder builder = new LuceneQueryBuilder(root,
71  		session, nsMappings, analyzer);
72  
73  	Query q = builder.createLuceneQuery();
74  	if (builder.exceptions.size() > 0) {
75  	    StringBuffer msg = new StringBuffer();
76  	    for (Iterator it = builder.exceptions.iterator(); it.hasNext(); ) {
77  		msg.append(it.next().toString()).append('\n');
78  	    }
79  	    throw new RepositoryException("Exception parsing query: " + msg.toString());
80  	}
81  	return q;
82      }

88      public Object visit(QueryRootNode node, Object data) {
89  	BooleanQuery root = new BooleanQuery();
90  	Query constraintQuery = (Query)node.getConstraintNode().accept(this, null);
91  	if (constraintQuery != null) {
92  	    root.add(constraintQuery, true, false);
93  	}
94  
95  	String[] props = node.getSelectProperties();
96  	for (int i = 0; i < props.length; i++) {
97  	    String prop = props[i];
98  	    try {
99  		prop = nsMappings.translatePropertyName(prop, session.getNamespaceResolver());
100 	    } catch (MalformedPathException e) {
101 		exceptions.add(e);
102 	    }
103 	    root.add(new MatchAllQuery(prop), true, false);
104 	}
105 
106 	TextsearchQueryNode textsearchNode = node.getTextsearchNode();
107 	if (textsearchNode != null) {
108 	    Query textsearch = (Query)textsearchNode.accept(this, null);
109 	    if (textsearch != null) {
110 		root.add(textsearch, true, false);
111 	    }
112 	}
113 
114 	Query wrapped = root;
115 	if (node.getLocationNode() != null) {
116 	    wrapped = (Query)node.getLocationNode().accept(this, root);
117 	}
118 
119 	return wrapped;
120     }

178 	} catch (MalformedPathException e) {

180 	}
181 	return new TermQuery(new Term(field, value));

188     public Object visit(TextsearchQueryNode node, Object data) {
189 	try {
190 	    org.apache.lucene.queryParser.QueryParser parser
191 		    = new org.apache.lucene.queryParser.QueryParser(FieldNames.FULLTEXT, analyzer);
192 	    parser.setOperator(org.apache.lucene.queryParser.QueryParser.DEFAULT_OPERATOR_AND);
193 	    // replace unescaped ' with " and escaped ' with just '
194 	    StringBuffer query = new StringBuffer();
195 	    String textsearch = node.getQuery();
196 	    boolean escaped = false;
197 	    for (int i = 0; i < textsearch.length(); i++) {
198 		if (textsearch.charAt(i) == '//') {
199 		    if (escaped) {
200 			query.append("////");
201 			escaped = false;
202 		    } else {
203 			escaped = true;
204 		    }
205 		} else if (textsearch.charAt(i) == '\'') {
206 		    if (escaped) {
207 			query.append('\'');
208 			escaped = false;
209 		    } else {
210 			query.append('\"');
211 		    }
212 		} else {
213 		    if (escaped) {
214 			query.append('//');
215 			escaped = false;
216 		    }
217 		    query.append(textsearch.charAt(i));
218 		}
219 	    }
220 	    return parser.parse(query.toString());
221 	} catch (ParseException e) {
222 	    exceptions.add(e);
223 	}
224 	return null;
225     }

239     public Object visit(RelationQueryNode node, Object data) {
240 	Query query;
241 	String stringValue;
242 	switch (node.getType()) {
243 	    case Constants.TYPE_DATE:
244 		stringValue = DateField.dateToString(node.getDateValue());
245 		break;
246 	    case Constants.TYPE_DOUBLE:
247 		stringValue = DoubleField.doubleToString(node.getDoubleValue());
248 		break;
249 	    case Constants.TYPE_LONG:
250 		stringValue = LongField.longToString(node.getLongValue());
251 		break;
252 	    case Constants.TYPE_STRING:
253 		stringValue = node.getStringValue();
254 		break;
255 	    default:
256 		throw new IllegalArgumentException("Unknown relation type: "
257 			+ node.getType());
258 	}
259 
260 	String field = node.getProperty();
261 	try {
262 	    field = nsMappings.translatePropertyName(node.getProperty(),
263 		    session.getNamespaceResolver());
264 	} catch (MalformedPathException e) {
265 	    exceptions.add(e);
266 	}
267 
268 	switch (node.getOperation()) {
269 	    case Constants.OPERATION_EQ:	// =
270 		query = new TermQuery(new Term(field, stringValue));
271 		break;
272 	    case Constants.OPERATION_GE:	// >=
273 		query = new RangeQuery(new Term(field, stringValue), null, true);
274 		break;
275 	    case Constants.OPERATION_GT:	// >
276 		query = new RangeQuery(new Term(field, stringValue), null, false);
277 		break;
278 	    case Constants.OPERATION_LE:	// <=
279 		query = new RangeQuery(null, new Term(field, stringValue), true);
280 		break;
281 	    case Constants.OPERATION_LIKE:	// LIKE
282 		query = new WildcardQuery(new Term(field, stringValue));
283 		break;
284 	    case Constants.OPERATION_LT:	// <
285 		query = new RangeQuery(null, new Term(field, stringValue), false);
286 		break;
287 	    case Constants.OPERATION_NE:	// !=
288 		BooleanQuery notQuery = new BooleanQuery();
289 		notQuery.add(new MatchAllQuery(field), false, false);
290 		notQuery.add(new TermQuery(new Term(field, stringValue)), false, true);
291 		query = notQuery;
292 		break;
293 	    default:
294 		throw new IllegalArgumentException("Unknown relation operation: "
295 			+ node.getOperation());
296 	}
297 	return query;
298     }
