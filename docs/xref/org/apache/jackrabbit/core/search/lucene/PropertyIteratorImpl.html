16  package org.apache.jackrabbit.core.search.lucene;

18  import org.apache.log4j.Logger;
19  
20  import javax.jcr.*;
21  import java.util.NoSuchElementException;

27  class PropertyIteratorImpl implements PropertyIterator {
28  
29      private static final Logger log = Logger.getLogger(PropertyIteratorImpl.class);
30  
31      private String[] props;
32  
33      private NodeIterator nodes;
34  
35      private Property next;
36  
37      private PropertyIterator currentProps;
38  
39      private long pos;
40  
41      PropertyIteratorImpl(String[] props, NodeIterator nodes) {
42  	this.nodes = nodes;
43  
44  	if (props != null && props.length > 0) {
45  	    this.props = props;
46  	}
47  	try {
48  	    fetchNext();
49  	} catch (RepositoryException e) {
50  	    // FIXME this is bad error handling!
51  	    log.error("Exception retrieving property: " + e.toString());
52  	}
53      }
54  
55      public Property nextProperty() {
56  	if (next == null) throw new NoSuchElementException();
57  	try {
58  	    Property tmp = next;
59  	    fetchNext();
60  	    pos++;
61  	    return tmp;
62  	} catch (RepositoryException e) {
63  	    log.error("Exception retrieving property: " + e.toString());
64  	    // FIXME this is bad error handling!
65  	    throw new NoSuchElementException();
66  	}
67      }
68  
69      public void skip(long skipNum) {
70  	while (skipNum-- > 0) {
71  	    next();
72  	}
73      }
74  
75      public long getSize() {
76  	return -1;
77      }
78  
79      public long getPos() {
80  	return pos;
81      }
82  
83      public void remove() {
84  	throw new UnsupportedOperationException("remove");
85      }
86  
87      public boolean hasNext() {
88  	return (next != null);
89      }
90  
91      public Object next() {
92  	return nextProperty();
93      }
94  
95      //--------------------< internal >------------------------------------------
96  
97      private void fetchNext() throws RepositoryException {
98  	next = null;
99  	if (currentProps == null) {
100 	    // try to get next PropertyIterator
101 	    if (nodes.hasNext()) {
102 		if (props != null) {
103 		    currentProps = new FilteredPropertyIterator(props, nodes.nextNode());
104 		} else {
105 		    currentProps = nodes.nextNode().getProperties();
106 		}
107 	    }
108 	}
109 
110 	if (currentProps != null) {
111 	    next = currentProps.nextProperty();
112 	    if (!currentProps.hasNext()) {
113 		// reset current iterator
114 		currentProps = null;
115 	    }
116 	}
117     }
118 }
