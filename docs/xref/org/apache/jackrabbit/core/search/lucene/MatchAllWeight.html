16  package org.apache.jackrabbit.core.search.lucene;
17  
18  import org.apache.lucene.search.*;
19  import org.apache.lucene.index.IndexReader;
20  
21  import java.io.IOException;
22  
23  /***
24   * This class implements the Weight calculation for the MatchAllQuery.
25   *
26   * @author Marcel Reutegger
27   * @version $Revision: 1.2 $, $Date: 2004/06/14 09:28:03 $
28   */
29  class MatchAllWeight implements Weight {
30  
31      /*** the MatchAllQuery */
32      private final Query query;
33  
34      private final String field;
35  
36      /*** the current Searcher instance */
37      private final Searcher searcher;
38  
39      /*** the weight value */
40      private float value;
41  
42      /*** doc frequency for this weight */
43      private float idf;
44  
45      /*** the query weight */
46      private float queryWeight;
47  
48      /***
49       *
50       * @param query
51       * @param searcher
52       */
53      MatchAllWeight(Query query, Searcher searcher, String field) {
54          this.query = query;
55          this.searcher = searcher;
56          this.field = field;
57      }
58  
59      /***
60       * @see Weight#getQuery()
61       */
62      public Query getQuery() {
63          return query;
64      }
65  
66      /***
67       * @see Weight#getValue()
68       */
69      public float getValue() {
70          return value;
71      }
72  
73      /***
74       * @see Weight#sumOfSquaredWeights()
75       */
76      public float sumOfSquaredWeights() throws IOException {
77          idf = searcher.getSimilarity().idf(searcher.maxDoc(), searcher.maxDoc()); // compute idf
78          queryWeight = idf * 1.0f; // boost         // compute query weight
79          return queryWeight * queryWeight;           // square it
80      }
81  
82      /***
83       * @see Weight#normalize(float)
84       */
85      public void normalize(float queryNorm) {
86          queryWeight *= queryNorm;                   // normalize query weight
87          value = queryWeight * idf;                  // idf for document
88      }
89  
90      /***
91       * @see Weight#scorer(org.apache.lucene.index.IndexReader)
92       */
93      public Scorer scorer(IndexReader reader) throws IOException {
94          return new MatchAllScorer(reader, this, field);
95      }
96  
97      /***
98       * @see Weight#explain(org.apache.lucene.index.IndexReader, int)
99       */
100     public Explanation explain(IndexReader reader, int doc) throws IOException {
101         return new Explanation(Similarity.getDefault().idf(reader.maxDoc(), reader.maxDoc()),
102                             "matchAll");
103     }
104 }
