75      public QueryImpl(SessionImpl session,
76  	      ItemManager itemMgr,
77  	      SearchManager searchMgr,
78  	      String absPath)
79  	    throws ItemNotFoundException, InvalidQueryException, RepositoryException {
80  
81      	this.session = session;
82  	this.itemMgr = itemMgr;
83  	this.searchMgr = searchMgr;
84  
85  	try {
86  	    Node query = null;
87  	    if (session.getRootNode().hasNode(absPath)) {
88  		query = session.getRootNode().getNode(absPath);
89  		// assert query has mix:referenceable
90  		query.getUUID();
91  		NodeType ntQuery = session.getWorkspace().getNodeTypeManager().getNodeType(NodeTypeRegistry.NT_QUERY.toJCRName(session.getNamespaceResolver()));
92  		if (query.getPrimaryNodeType().equals(ntQuery));
93  	    } else {
94  		throw new ItemNotFoundException(absPath);
95  	    }
96  
97  	    path = Path.create(absPath, session.getNamespaceResolver(), true);
98  
99  	    statement = query.getProperty(PROP_STATEMENT.toJCRName(session.getNamespaceResolver())).getString();
100 	    language = query.getProperty(PROP_LANGUAGE.toJCRName(session.getNamespaceResolver())).getString();
101 
102 	    // parse query according to language
103 	    // build query tree and pass to QueryImpl
104 	    QueryRootNode root = QueryParser.parse(statement, language/*, session.getNamespaceResolver()*/);
105 	    this.root = root;
106 	} catch (NoPrefixDeclaredException e) {
107 	    throw new InvalidQueryException(e.getMessage(), e);
108 	} catch (MalformedPathException e) {
109 	    throw new ItemNotFoundException(absPath, e);
110 	}
111     }

151     public void save(String absPath)
152 	    throws ItemExistsException,
153 	    PathNotFoundException,
154 	    ConstraintViolationException,
155 	    RepositoryException {
156 	try {
157 	    NamespaceResolver resolver = session.getNamespaceResolver();
158 	    Path p = Path.create(absPath, resolver, true);
159 	    if (!p.isAbsolute()) {
160 		throw new RepositoryException(absPath + " is not absolut");
161 	    }
162 	    if (!session.getRootNode().hasNode(p.getAncestor(1).toJCRPath(resolver))) {
163 		throw new PathNotFoundException(p.getAncestor(1).toJCRPath(resolver));
164 	    }
165 	    if (session.getRootNode().hasNode(p.toJCRPath(resolver))) {
166 		throw new ItemExistsException(p.toJCRPath(resolver));
167 	    }
168 	    Node queryNode = session.getRootNode().addNode(p.toJCRPath(resolver),
169 		    NodeTypeRegistry.NT_QUERY.toJCRName(resolver));
170 	    // set 
171 	    queryNode.setProperty(PROP_LANGUAGE.toJCRName(resolver), language);
172 	    queryNode.setProperty(PROP_STATEMENT.toJCRName(resolver), statement);
173 	    // add mixin referenceable
174 	    queryNode.addMixin(NodeTypeRegistry.MIX_REFERENCEABLE.toJCRName(resolver));
175 	    // FIXME do anything else?
176 	    queryNode.save();
177 
178 	} catch (MalformedPathException e) {
179 	    throw new RepositoryException(e.getMessage(), e);
180 	} catch (NoPrefixDeclaredException e) {
181 	    throw new RepositoryException(e.getMessage(), e);
182 	}
183 	session.getRootNode().getNode(absPath);
184     }
