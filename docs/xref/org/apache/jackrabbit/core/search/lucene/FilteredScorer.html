16  package org.apache.jackrabbit.core.search.lucene;
17  
18  import org.apache.lucene.search.Scorer;
19  import org.apache.lucene.search.Explanation;
20  import org.apache.lucene.search.HitCollector;
21  import org.apache.lucene.search.Similarity;
22  import org.apache.lucene.index.IndexReader;
23  import org.apache.jackrabbit.core.search.lucene.PackageFilter;
24  
25  import java.io.IOException;
26  import java.util.BitSet;
27  
28  /***
29   * Implements a <code>Scorer</code> which only scores hits,
30   * that are not filtered out by the {@link org.apache.jackrabbit.core.search.lucene.PackageFilter}.
31   *
32   * @version $Revision: 1.1 $, $Date: 2004/01/23 15:55:40 $
33   * @author Marcel Reutegger
34   * @since gumbear
35   */
36  class FilteredScorer extends Scorer {
37  
38      /*** The <code>Scorer</code> of the real lucene query */
39      private Scorer delegatee;
40  
41      /*** The {@link org.apache.jackrabbit.core.search.lucene.PackageFilter} for filtering the hits */
42      private PackageFilter filter;
43  
44      /*** The <code>IndexReader</code> for reading from the search index */
45      private IndexReader reader;
46  
47      /*** The <code>HitCollector</code> with filtering facility */
48      private FilteredHitCollector collector;
49  
50      /***
51       * Creates a new <code>FilteredScorer</code> based on a concrete
52       * <code>Scorer</code> using a {@link org.apache.jackrabbit.core.search.lucene.PackageFilter}.
53       * @param delegatee
54       * @param filter
55       * @param reader
56       */
57      FilteredScorer(Scorer delegatee, PackageFilter filter, IndexReader reader) {
58  	super(Similarity.getDefault());
59  	this.delegatee = delegatee;
60  	this.filter = filter;
61  	this.reader = reader;
62      }
63  
64      /***
65       * @see Scorer#explain
66       */
67      public Explanation explain(int doc) throws IOException {
68  	return delegatee.explain(doc);
69  
70      }
71  
72      /***
73       * Returns a {@link FilteredHitCollector} which filters
74       * the collected hits with a {@link org.apache.jackrabbit.core.search.lucene.PackageFilter}.
75       * @param hc the <code>HitCollector</code> from the underlying
76       *   lucene query.
77       * @param maxDoc collect hits until <code>maxDoc</code> has reached.
78       * @throws IOException if an error occurs while collecting hits.
79       *   e.g. while reading from the search index.
80       */
81      public void score(HitCollector hc, int maxDoc) throws IOException {
82  	if (collector == null) {
83  	    collector = new FilteredHitCollector();
84  	}
85  	collector.setDelegatee(hc);
86  	delegatee.score(collector, maxDoc);
87      }
88  
89      //---------------------< inner class >---------------------------
90  
91      private class FilteredHitCollector extends HitCollector {
92  
93  	/*** The unfiltered <code>HitCollector</code>. */
94  	private HitCollector delegateeCollector;
95  
96  	/*** The filter */
97  	private BitSet filterBitSet;
98  
99  	/***
100 	 * Creates a new <code>FilteredHitCollector</code>.
101 	 * @throws IOException if an error occurs while
102 	 *   calculating the filter.
103 	 */
104 	FilteredHitCollector() throws IOException {
105 	    filterBitSet = filter.bits(reader);
106 	}
107 
108 	/***
109 	 * Sets the currently used unfiltered <code>HitCollector</code>.
110 	 * @param hc the new <code>HitCollector</code>.
111 	 */
112 	private void setDelegatee(HitCollector hc) {
113 	    delegateeCollector = hc;
114 	}
115 
116 	/***
117 	 * Collects the document with number <code>doc</code>.
118 	 * This hit collector first applies the filter on the
119 	 * document and only passes the collect call to the
120 	 * underlying <code>HitCollector</code> if the filter
121 	 * allows the document.
122 	 * @param doc the document in question for filtering.
123 	 * @param score the score for this document.
124 	 */
125 	public void collect(int doc, float score) {
126 	    // only forward the collect call if the filter
127 	    // allows the document.
128 	    if (filterBitSet.get(doc)) {
129 		delegateeCollector.collect(doc, score);
130 	    }
131 	}
132     }
133 }
