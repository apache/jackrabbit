16  package org.apache.jackrabbit.core.search.lucene;
17  
18  import org.apache.lucene.search.*;
19  import org.apache.lucene.index.IndexReader;
20  import org.apache.lucene.index.TermEnum;
21  import org.apache.lucene.index.Term;
22  import org.apache.lucene.index.TermDocs;
23  
24  import java.io.IOException;
25  import java.util.BitSet;
26  
27  /***
28   * The MatchAllScorer implements a Scorer that scores / collects all
29   * documents in the index that match an optional field / atom-filter.
30   * In case there are no filters, this MatchAllScores simply collects
31   * all documents in the index that are not marked as deleted.
32   *
33   * @author Marcel Reutegger
34   * @version $Revision: 1.2 $, $Date: 2004/06/15 13:52:13 $
35   */
36  class MatchAllScorer extends Scorer {
37  
38      /*** current doc number */
39      private int docNo = 0;
40  
41      /*** IndexReader giving access to index */
42      private IndexReader reader;
43  
44      /*** Weight associated with this Scorer */
45      private Weight weight;
46  
47      /*** The field to match */
48      private String field;
49  
50      /*** BitSet filtering documents without content is specified field */
51      private BitSet docFilter;
52  
53      /*** Explanation object. the same for all docs */
54      private Explanation MATCH_EXPL;
55  
56      /***
57       * Creates a new MatchAllScorer.
58       *
59       * @param reader the IndexReader
60       * @param weight associated Weight for this Scorer
61       * @param field the field name to match.
62       * @throws IOException if an error occurs while collecting hits.
63       *   e.g. while reading from the search index.
64       */
65      MatchAllScorer(IndexReader reader, Weight weight, String field)
66  	    throws IOException {
67          super(Similarity.getDefault());
68          this.reader = reader;
69          this.weight = weight;
70          this.field = field;
71          MATCH_EXPL
72              = new Explanation(Similarity.getDefault().idf(reader.maxDoc(),
73                      reader.maxDoc()),
74                      "matchAll");
75          calculateDocFilter();
76      }
77  
78      /***
79       * Scores documents until <code>maxDoc</code> has reached.
80       *
81       * @param hc the <code>HitCollector</code> from the underlying
82       *   lucene query.
83       * @param maxDoc collect hits until <code>maxDoc</code> has reached.
84       */
85      public void score(HitCollector hc, int maxDoc) {
86          float score = getSimilarity().tf(1) * weight.getValue();
87          while (docNo < maxDoc) {
88              if (!reader.isDeleted(docNo)) {
89  		// check docFilter
90  		if (docFilter.get(docNo)) {
91  		    hc.collect(docNo, score);
92  		}
93              }
94              docNo++;
95          }
96      }
97  
98      /***
99       * @see Scorer#explain
100      */
101     public Explanation explain(int doc) {
102         return MATCH_EXPL;
103     }
104 
105     /***
106      * Calculates a BitSet filter that includes all the nodes
107      * that have content in properties according to the field name
108      * passed in the constructor of this MatchAllScorer.
109      *
110      * @throws IOException if an error occurs while reading from
111      *    the search index.
112      */
113     private void calculateDocFilter() throws IOException {
114         docFilter = new BitSet(reader.maxDoc());
115 	// we match all terms
116 	TermEnum terms = null;
117 	try {
118 	    terms = reader.terms(new Term(field, ""));
119 	    while (terms.term() != null && terms.term().field() == field) {
120 		TermDocs termDocs = null;
121 		try {
122 		    termDocs = reader.termDocs(terms.term());
123 		    while (termDocs.next()) {
124 			docFilter.set(termDocs.doc());
125 		    }
126 		} finally {
127 		    if (termDocs != null) {
128 			termDocs.close();
129 		    }
130 		}
131 		terms.next();
132 	    }
133 	} finally {
134 	    if (terms != null) {
135 		terms.close();
136 	    }
137 	}
138     }
139 }
