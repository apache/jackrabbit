16  package org.apache.jackrabbit.core.search.lucene;

18  import org.apache.lucene.search.Query;

20  import org.apache.lucene.search.Searcher;

22  /***
23   * Specialized query that returns / scores all pages in the search index.
24   * <p>Use this Query to perform a match '*'.
25   *
26   * @author Marcel Reutegger
27   * @version $Revision: 1.2 $, $Date: 2004/06/14 09:28:03 $
28   */
29  class MatchAllQuery extends Query {
30  
31      private final String field;
32  
33      /***
34       * Creates a new <code>MatchAllQuery</code> .
35       * <p>
36       *
37       * @param field the field name.
38       * @throws NullPointerException if <code>field</code> is null.
39       */
40      MatchAllQuery(String field) {
41          if (field == null) throw new NullPointerException("field");
42          this.field = field.intern();
43      }
44  
45      /***
46       * Returns the <code>Weight</code> for this Query.
47       * @param searcher the current searcher.
48       * @return the <code>Weight</code> for this Query.
49       */
50      protected Weight createWeight(Searcher searcher) {
51          return new MatchAllWeight(this, searcher, field);
52      }
53  
54      /***
55       * Returns the String "*".
56       * @param field default field for the query.
57       * @return the String "*".
58       */
59      public String toString(String field) {
60          return "*";
61      }
62  }
