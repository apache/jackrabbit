16  package org.apache.jackrabbit.core.search.lucene;
17  
18  import org.apache.lucene.search.Filter;
19  import org.apache.lucene.index.IndexReader;
20  
21  import java.util.BitSet;
22  import java.io.IOException;
23  
24  /***
25   * Implements a fulltext filter, based on a {@link PathFilter}.
26   *
27   * @version $Revision: 1.3 $, $Date: 2004/06/15 13:52:13 $
28   * @author Marcel Reutegger
29   */
30  class PackageFilter extends Filter {
31  
32      /*** the <code>BitSet</code> representing the filter */
33      private BitSet filter = null;
34  
35      /*** number of bits set true */
36      private int bitCount = 0;
37  
38      /*** the {@link PathFilter} defining the filter range */
39      private final PathFilter pathFilter;
40  
41      /***
42       * Creates a {@link PackageFilter} for the {@link PathFilter}
43       * <code>filter</code>.<br>
44       *
45       * @param filter the {@link PathFilter} defining the filter range
46       */
47      public PackageFilter(PathFilter filter) {
48  	pathFilter = filter;
49      }
50  
51      /***
52       * Returns a <code>BitSet</code> where bits are set true for documents
53       * which are contained in the {@link PathFilter} provided in the
54       * Constructor of this class.
55       *
56       * @param reader the <code>IndexReader</code> of the search index.
57       * @return a BitSet with true for documents which should be permitted in search
58       *     results, and false for those that should not.
59       * @throws IOException if an error occurs while reading from the search index.
60       */
61      public synchronized BitSet bits(IndexReader reader) throws IOException {
62  	// check if previously calculated
63  	if (filter == null) {
64  	    filter = new BitSet(reader.maxDoc());
65  
66  	    // Iterate over all docs
67  	    for (int i = 0; i < reader.maxDoc(); i++) {
68  		if (!reader.isDeleted(i)) {
69  		    // check if document is in ContentPackage
70  		    if (pathFilter.includes(reader.document(i).getField(FieldNames.PATH).stringValue())) {
71  			filter.set(i);
72  			bitCount++;
73  		    }
74  		}
75  	    }
76  	}
77  	return filter;
78      }
79  
80      /***
81       * Returns true if this PackageFilter blocks all pages, false
82       * otherwise. Please note that this method returns false when
83       * this PackageFilter was never called with a reader before.
84       * See {@link #bits}.
85       * @return true if this <code>PackageFilter</code> blocks all pages.
86       */
87      public boolean blocksAll() {
88  	return (filter != null && bitCount == 0);
89      }
90  }
