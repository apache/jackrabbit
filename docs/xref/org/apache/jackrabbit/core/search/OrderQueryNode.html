16  package org.apache.jackrabbit.core.search;
17  
18  /***
19   * Implements a query node that defines the order of nodes according to the
20   * values of properties.
21   *
22   * @author Marcel Reutegger
23   * @version $Revision:  $, $Date:  $
24   */
25  public class OrderQueryNode extends QueryNode {
26  
27      /*** The name of the properties to order */
28      private String[] properties;
29  
30      /*** Flag indicating whether nodes are ordered ascending or descending */
31      private boolean ascending;
32  
33      /***
34       * Creates a new <code>OrderQueryNode</code> with a reference to a parent
35       * node and sort properties.
36       *
37       * @param parent     the parent node of this query node.
38       * @param properties the names of the properties to sort the result nodes.
39       * @param asc        if <code>true</code> result nodes are orderd ascending;
40       *                   otherwise descending.
41       */
42      public OrderQueryNode(QueryNode parent, String[] properties, boolean asc) {
43  	super(parent);
44  	this.properties = properties;
45  	this.ascending = asc;
46      }
47  
48  
49      /***
50       * @see QueryNode#accept(org.apache.jackrabbit.core.search.QueryNodeVisitor, java.lang.Object)
51       */
52      public Object accept(QueryNodeVisitor visitor, Object data) {
53  	return visitor.visit(this, data);
54      }
55  
56      /***
57       * Returns <code>true</code> if result nodes should be orderd ascending.
58       * If <code>false</code> result nodes are ordered descending.
59       * @return the value of the ascending property.
60       */
61      public boolean isAscending() {
62  	return ascending;
63      }
64  
65      /***
66       * Returns a String array that contains the name of the properties
67       * to sort the result nodes.
68       * @return names of order properties.
69       */
70      public String[] getOrderByProperties() {
71  	return properties;
72      }
73  
74      /***
75       * Returns a JCRQL representation for this query node.
76       * @return a JCRQL representation for this query node.
77       */
78      public String toJCRQLString() {
79  	StringBuffer sb = new StringBuffer("ORDER BY");
80  	if (properties.length > 0) {
81  	    String comma = "";
82  	    for (int i = 0; i < properties.length; i++) {
83  		sb.append(comma).append(" ");
84  		sb.append(properties[i]);
85  		comma = ",";
86  	    }
87  	} else {
88  	    sb.append(" SCORE");
89  	}
90  	if (ascending) {
91  	    // FIXME really default to descending?
92  	    sb.append(" ASCENDING");
93  	}
94  	return sb.toString();
95      }
96  
97      /***
98       * Returns an XPath representation for this query node.
99       * @return an XPath representation for this query node.
100      */
101     public String toXPathString() {
102 	return "";
103     }
104 }
