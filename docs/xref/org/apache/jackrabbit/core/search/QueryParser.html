16  package org.apache.jackrabbit.core.search;
17  
18  import org.apache.jackrabbit.core.search.jcrql.JCRQLQueryBuilder;
19  
20  import javax.jcr.query.Query;
21  import javax.jcr.query.InvalidQueryException;
22  
23  /***
24   * This class acts as the central entry point for parsing query statements from
25   * different query syntaxes into a query tree.
26   *
27   * @author Marcel Reutegger
28   * @version $Revision:  $, $Date:  $
29   */
30  public class QueryParser {
31  
32      /***
33       * This class cannot be instanciated.
34       */
35      private QueryParser() {
36      }
37  
38      /***
39       * Parses a query <code>statement</code> according to a query
40       * <code>language</code> into a query tree.
41       * <p/>
42       * <code>language</code> must be one of: {@link javax.query.Query#JCRQL},
43       * {@link javax.query.Query#XPATH_DOCUMENT_VIEW}, {@link
44       * javax.query.Query#XPATH_SYSTEM_VIEW}.
45       *
46       * @param statement the query statement.
47       * @param language  the language of the query statement.
48       * @return the root node of the generated query tree.
49       *
50       * @throws InvalidQueryException if an error occurs while parsing the
51       *                               statement.
52       */
53      public static QueryRootNode parse(String statement, String language)
54  	    throws InvalidQueryException {
55  
56  	if (language == Query.JCRQL) {
57  	    return JCRQLQueryBuilder.createQuery(statement);
58  	} else {
59  	    throw new InvalidQueryException("unknown language");
60  	}
61      }
62  
63  }
