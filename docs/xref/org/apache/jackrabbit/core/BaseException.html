16  package org.apache.jackrabbit.core;
17  
18  import java.io.PrintStream;
19  import java.io.PrintWriter;
20  
21  /***
22   * The abstract class <code>BaseException</code> serves as the base class
23   * for all exceptions that are specific to this JCR implementation and that
24   * are not derived from <code>javax.jcr.RepositoryException</code>.
25   *
26   * @author Stefan Guggisberg
27   * @version $Revision: 1.5 $, $Date: 2004/09/01 15:14:27 $
28   */
29  public abstract class BaseException extends Exception {
30      /***
31       * Root failure cause
32       */
33      protected Throwable rootCause;
34  
35      /***
36       * Constructs a new instance of this class with <code>null</code> as its
37       * detail message.
38       */
39      public BaseException() {
40  	super();
41      }
42  
43      /***
44       * Constructs a new instance of this class with the specified detail
45       * message.
46       *
47       * @param message the detail message. The detail message is saved for
48       *                later retrieval by the {@link #getMessage()} method.
49       */
50      public BaseException(String message) {
51  	super(message);
52      }
53  
54      /***
55       * Constructs a new instance of this class with the specified detail
56       * message and root cause.
57       *
58       * @param message   the detail message. The detail message is saved for
59       *                  later retrieval by the {@link #getMessage()} method.
60       * @param rootCause root failure cause
61       */
62      public BaseException(String message, Throwable rootCause) {
63  	super(message);
64  	this.rootCause = rootCause;
65      }
66  
67      /***
68       * Constructs a new instance of this class with the specified root cause.
69       *
70       * @param rootCause root failure cause
71       */
72      public BaseException(Throwable rootCause) {
73  	super();
74  	this.rootCause = rootCause;
75      }
76  
77      /***
78       * Returns the detail message, including the message from the nested
79       * exception if there is one.
80       *
81       * @return the detail message (which may be <code>null</code>).
82       */
83      public String getMessage() {
84  	String s = super.getMessage();
85  	if (rootCause == null) {
86  	    return s;
87  	} else {
88  	    String s2 = rootCause.getMessage();
89  	    return s == null ? s2 : s + ": " + s2;
90  	}
91      }
92  
93      /***
94       * Creates a localized description of this exception.
95       * Subclasses may override this method in order to produce a
96       * locale-specific message. For subclasses that do not override this
97       * method, the default implementation returns the same result as
98       * <code>getMessage()</code>.
99       *
100      * @return The localized description of this exception.
101      */
102     public String getLocalizedMessage() {
103 	return getMessage();
104     }
105 
106     /***
107      * Returns the cause of this exception or <code>null</code> if the
108      * cause is nonexistent or unknown. (The cause is the throwable that
109      * caused this exception to get thrown.)
110      *
111      * @return the cause of this exception or <code>null</code> if the
112      *         cause is nonexistent or unknown.
113      */
114     public Throwable getCause() {
115 	return rootCause;
116     }
117 
118     /***
119      * Prints this <code>RepositoryException</code> and its backtrace to the
120      * standard error stream.
121      */
122     public void printStackTrace() {
123 	printStackTrace(System.err);
124     }
125 
126     /***
127      * Prints this <code>RepositoryException</code> and its backtrace to the
128      * specified print stream.
129      *
130      * @param s <code>PrintStream</code> to use for output
131      */
132     public void printStackTrace(PrintStream s) {
133 	synchronized (s) {
134 	    super.printStackTrace(s);
135 	    if (rootCause != null) {
136 		rootCause.printStackTrace(s);
137 	    }
138 	}
139     }
140 
141     /***
142      * Prints this <code>RepositoryException</code> and its backtrace to
143      * the specified print writer.
144      *
145      * @param s <code>PrintWriter</code> to use for output
146      */
147     public void printStackTrace(PrintWriter s) {
148 	synchronized (s) {
149 	    super.printStackTrace(s);
150 	    if (rootCause != null) {
151 		rootCause.printStackTrace(s);
152 	    }
153 	}
154     }
155 }
