38      private static Logger log = Logger.getLogger(DocViewImportHandler.class);

40      private Stack parents;
41      private SessionImpl session;
42  
43      DocViewImportHandler(NodeImpl importTargetNode, SessionImpl session) {
44  	this.session = session;
45  	parents = new Stack();
46  
47  	parents.push(importTargetNode);
48      }

51      /***
52       * @see ContentHandler#startElement(String, String, String, Attributes)
53       */
54      public void startElement(String namespaceURI, String localName, String qName, Attributes atts) throws SAXException {
55  	try {
56  	    QName nodeName;
57  	    if (qName == null || "".equals(qName)) {
58  		nodeName = new QName(namespaceURI, localName);
59  	    } else {
60  		try {
61  		    nodeName = QName.fromJCRName(qName, session.getNamespaceResolver());
62  		} catch (IllegalNameException ine) {
63  		    throw new SAXException("illegal node name: " + qName, ine);
64  		} catch (UnknownPrefixException upe) {
65  		    throw new SAXException("illegal node name: " + qName, upe);
66  		}
67  	    }
68  
69  	    // @todo how should 'system' properties be handled in document view (e.g. jcr:primaryType,jcr:mixinTypes, jcr:uuid)?
70  	    NodeImpl currentParent = (NodeImpl) parents.peek();
71  	    currentParent = (NodeImpl) currentParent.addNode(nodeName, NodeTypeRegistry.NT_UNSTRUCTURED);
72  	    parents.push(currentParent);
73  
74  	    // properties
75  	    for (int i = 0; i < atts.getLength(); i++) {
76  		QName propName;
77  		if (atts.getQName(i) == null || "".equals(atts.getQName(i))) {
78  		    propName = new QName(atts.getURI(i), atts.getLocalName(i));
79  		} else {
80  		    try {
81  			propName = QName.fromJCRName(atts.getQName(i), session.getNamespaceResolver());
82  		    } catch (IllegalNameException ine) {
83  			throw new SAXException("illegal property name: " + atts.getQName(i), ine);
84  		    } catch (UnknownPrefixException upe) {
85  			throw new SAXException("illegal property name: " + atts.getQName(i), upe);
86  		    }
87  		}
88  		StringValue val = new StringValue(atts.getValue(i));
89  		currentParent.setProperty(propName, val);
90  	    }
91  	} catch (RepositoryException re) {
92  	    throw new SAXException(re);
93  	}
94      }
95  
96      /***
97       * @see ContentHandler#characters(char[], int, int)
98       */
99      public void characters(char[] ch, int start, int length) throws SAXException {
100 	// character data in document view:
101 	// store as jcr:xmlcharacters property of jcr:xmltext node
102 	// (need to store as node in order to maintain ordering)
103 	try {
104 	    NodeImpl currentParent = (NodeImpl) parents.peek();
105 	    NodeImpl txtNode = (NodeImpl) currentParent.addNode(DocViewSAXEventGenerator.NODENAME_XMLTEXT);
106 	    StringValue val = new StringValue(new String(ch, start, length));
107 	    txtNode.setProperty(DocViewSAXEventGenerator.PROPNAME_XMLCHARACTERS, val);
108 	} catch (RepositoryException re) {
109 	    throw new SAXException(re);
110 	}
111     }

113     /***
114      * @see ContentHandler#endElement(String, String, String)
115      */
116     public void endElement(String namespaceURI, String localName, String qName) throws SAXException {
117 	parents.pop();
118     }
