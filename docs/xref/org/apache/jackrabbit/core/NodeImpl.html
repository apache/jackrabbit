77      protected synchronized ItemState getOrCreateTransientItemState()
78  	    throws RepositoryException {
79  	if (!isTransient()) {
80  	    try {
81  		// make transient (copy-on-write)
82  		NodeState transientState =
83  			itemStateMgr.createTransientNodeState((NodeState) state, ItemState.STATUS_EXISTING_MODIFIED);
84  		// remove listener on persistent state
85  		state.removeListener(this);
86  		// add listener on transient state
87  		transientState.addListener(this);
88  		// replace persistent with transient state
89  		state = transientState;
90  	    } catch (ItemStateException ise) {
91  		String msg = "failed to create transient state";
92  		log.error(msg, ise);
93  		throw new RepositoryException(msg, ise);
94  	    }
95  	}
96  	return state;
97      }

209     protected synchronized PropertyImpl createChildProperty(QName name, int type, PropertyDefImpl def)
210 	    throws RepositoryException {
211 	// check for name collisions with existing child nodes
212 	if (((NodeState) state).hasChildNodeEntry(name)) {
213 	    String msg = "there's already a child node with name " + name;
214 	    log.error(msg);
215 	    throw new RepositoryException(msg);
216 	}
217 
218 	String parentUUID = ((NodeState) state).getUUID();
219 
220 	// create a new property state
221 	PropertyState propState = null;
222 	try {
223 	    propState = itemStateMgr.createTransientPropertyState(parentUUID, name, ItemState.STATUS_NEW);
224 	    propState.setType(type);
225 	    propState.setDefinitionId(new PropDefId(def.unwrap()));
226 	    // compute system generated values if necessary
227 	    InternalValue[] genValues = computeSystemGeneratedPropertyValues(name, def);
228 	    if (genValues != null) {
229 		propState.setValues(genValues);
230 	    } else if (def.getDefaultValues() != null) {
231 		Value[] vals = def.getDefaultValues();
232 		int length = (def.isMultiple() ? vals.length : 1);
233 		InternalValue[] defVals = new InternalValue[length];
234 		for (int i = 0; i < length; i++) {
235 		    defVals[i] = InternalValue.create(vals[i], session.getNamespaceResolver());
236 		}
237 		propState.setValues(defVals);
238 	    }
239 	} catch (ItemStateException ise) {
240 	    String msg = "failed to add property " + name + " to " + safeGetJCRPath();
241 	    log.error(msg, ise);
242 	    throw new RepositoryException(msg, ise);
243 	}
244 
245 	// create Property instance wrapping new property state
246 	PropertyImpl prop = itemMgr.createPropertyInstance(propState, def);
247 
248 	// modify the state of 'this', i.e. the parent node
249 	NodeState thisState = (NodeState) getOrCreateTransientItemState();
250 	// add new property entry
251 	thisState.addPropertyEntry(name);
252 
253 	return prop;
254     }

256     protected synchronized NodeImpl createChildNode(QName name, NodeDefImpl def,
257 						    NodeTypeImpl nodeType, String uuid)
258 	    throws RepositoryException {
259 	String parentUUID = ((NodeState) state).getUUID();
260 	// create a new node state
261 	NodeState nodeState = null;
262 	try {
263 	    if (uuid == null) {
264 		uuid = UUID.randomUUID().toString();	// version 4 uuid
265 	    }
266 	    nodeState = itemStateMgr.createTransientNodeState(uuid, nodeType.getQName(), parentUUID, ItemState.STATUS_NEW);
267 	    nodeState.setDefinitionId(new NodeDefId(def.unwrap()));
268 	} catch (ItemStateException ise) {
269 	    String msg = "failed to add child node " + name + " to " + safeGetJCRPath();
270 	    log.error(msg, ise);
271 	    throw new RepositoryException(msg, ise);
272 	}
273 
274 	// create Node instance wrapping new node state
275 	NodeImpl node;
276 	try {
277 	    node = itemMgr.createNodeInstance(nodeState, def);
278 	} catch (RepositoryException re) {
279 	    // something went wrong
280 	    itemStateMgr.disposeTransientItemState(nodeState);
281 	    // re-throw
282 	    throw re;
283 	}
284 
285 	// modify the state of 'this', i.e. the parent node
286 	NodeState thisState = (NodeState) getOrCreateTransientItemState();
287 	// add new child node entry
288 	thisState.addChildNodeEntry(name, nodeState.getUUID());
289 
290 	// add 'auto-create' properties defined in node type
291 	PropertyDef[] pda = nodeType.getAutoCreatePropertyDefs();
292 	for (int i = 0; i < pda.length; i++) {
293 	    PropertyDefImpl pd = (PropertyDefImpl) pda[i];
294 	    node.createChildProperty(pd.getQName(), pd.getRequiredType(), pd);
295 	}
296 
297 	// recursively add 'auto-create' child nodes defined in node type
298 	NodeDef[] nda = nodeType.getAutoCreateNodeDefs();
299 	for (int i = 0; i < nda.length; i++) {
300 	    NodeDefImpl nd = (NodeDefImpl) nda[i];
301 	    node.createChildNode(nd.getQName(), nd, (NodeTypeImpl) nd.getDefaultPrimaryType(), null);
302 	}
303 
304 	return node;
305     }

338     protected void removeChildNode(QName nodeName, int index) throws RepositoryException {
339 	// modify the state of 'this', i.e. the parent node
340 	NodeState thisState = (NodeState) getOrCreateTransientItemState();
341 	index = (index == 0) ? 1 : index;
342 	NodeState.ChildNodeEntry entry = thisState.getChildNodeEntry(nodeName, index);
343 	if (entry == null) {
344 	    String msg = "failed to remove child " + nodeName + " of " + safeGetJCRPath();
345 	    log.error(msg);
346 	    throw new RepositoryException(msg);
347 	}
348 
349 	NodeId childId = new NodeId(entry.getUUID());
350 	NodeImpl childNode = (NodeImpl) itemMgr.getItem(childId);
351 	// notify target of removal/unlink
352 	childNode.onUnlink(thisState);
353 
354 	// remove child entry
355 	if (!thisState.removeChildNodeEntry(nodeName, index)) {
356 	    String msg = "failed to remove child " + nodeName + " of " + safeGetJCRPath();
357 	    log.error(msg);
358 	    throw new RepositoryException(msg);
359 	}
360     }

378     protected void onUnlink(NodeState parentState) throws RepositoryException {
379 	// modify the state of 'this', i.e. the target node
380 	NodeState thisState = (NodeState) getOrCreateTransientItemState();
381 
382 	// check if this node would be orphaned after unlinking it from parent
383 	ArrayList parentUUIDs = new ArrayList(thisState.getParentUUIDs());
384 	parentUUIDs.remove(parentState.getUUID());
385 	boolean orphaned = parentUUIDs.isEmpty();
386 
387 	if (orphaned) {
388 	    // remove child nodes (recursive)
389 	    // use temp array to avoid ConcurrentModificationException
390 	    ArrayList tmp = new ArrayList(thisState.getChildNodeEntries());
391 	    // remove from tail to avoid problems with same-name siblings
392 	    for (int i = tmp.size() - 1; i >= 0; i--) {
393 		NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) tmp.get(i);
394 		removeChildNode(entry.getName(), entry.getIndex());
395 	    }
396 
397 	    // remove properties
398 	    // use temp array to avoid ConcurrentModificationException
399 	    tmp = new ArrayList(thisState.getPropertyEntries());
400 	    // remove from tail to avoid problems with same-name siblings
401 	    for (int i = tmp.size() - 1; i >= 0; i--) {
402 		NodeState.PropertyEntry entry = (NodeState.PropertyEntry) tmp.get(i);
403 		removeChildProperty(entry.getName());
404 	    }
405 	}
406 
407 	// now actually do unlink this node from specified parent node
408 	// (i.e. remove uuid of parent node from this node's parent list)
409 	thisState.removeParentUUID(parentState.getUUID());
410 
411 	if (orphaned) {
412 	    // remove this node
413 	    itemMgr.removeItem(id);
414 	}
415     }

423     protected NodeImpl internalAddNode(String relPath, NodeTypeImpl nodeType,
424 				       String uuid)
425 	    throws ItemExistsException, PathNotFoundException,
426 	    ConstraintViolationException, RepositoryException {
427 	Path nodePath;
428 	QName nodeName;
429 	Path parentPath;
430 	try {
431 	    nodePath = Path.create(getPrimaryPath(), relPath, session.getNamespaceResolver(), true);
432 	    if (nodePath.getNameElement().getIndex() != 0) {
433 		String msg = "illegal subscript specified: " + nodePath;
434 		log.error(msg);
435 		throw new RepositoryException(msg);
436 	    }
437 	    nodeName = nodePath.getNameElement().getName();
438 	    parentPath = nodePath.getAncestor(1);
439 	} catch (MalformedPathException e) {
440 	    String msg = "failed to resolve path " + relPath + " relative to " + safeGetJCRPath();
441 	    log.error(msg, e);
442 	    throw new RepositoryException(msg, e);
443 	}
444 
445 	NodeImpl parentNode;
446 	try {
447 	    Item parent = itemMgr.getItem(parentPath);
448 	    if (!parent.isNode()) {
449 		String msg = "cannot add a node to property " + parentPath;
450 		log.error(msg);
451 		throw new ConstraintViolationException(msg);
452 	    }
453 	    parentNode = (NodeImpl) parent;
454 	} catch (AccessDeniedException ade) {
455 	    throw new PathNotFoundException(relPath);
456 	}
457 
458 	// delegate the creation of the child node to the parent node
459 	return parentNode.internalAddChildNode(nodeName, nodeType, uuid);
460     }

467     protected NodeImpl internalAddChildNode(QName nodeName, NodeTypeImpl nodeType, String uuid)
468 	    throws ItemExistsException, ConstraintViolationException, RepositoryException {
469 	Path nodePath;
470 	try {
471 	    nodePath = Path.create(getPrimaryPath(), nodeName, true);
472 	} catch (MalformedPathException e) {
473 	    // should never happen
474 	    String msg = "internal error: invalid path " + safeGetJCRPath();
475 	    log.error(msg, e);
476 	    throw new RepositoryException(msg, e);
477 	}
478 
479 	NodeDefImpl def;
480 	try {
481 	    def = getApplicableChildNodeDef(nodeName, nodeType == null ? null : nodeType.getQName());
482 	} catch (RepositoryException re) {
483 	    String msg = "no definition found in parent node's node type for new node";
484 	    log.error(msg, re);
485 	    throw new ConstraintViolationException(msg, re);
486 	}
487 	if (nodeType == null) {
488 	    // use default node type
489 	    nodeType = (NodeTypeImpl) def.getDefaultPrimaryType();
490 	}
491 
492 	// check for name collisions
493 	try {
494 	    ItemImpl item = itemMgr.getItem(nodePath);
495 	    if (!item.isNode()) {
496 		// there's already a property with that name
497 		throw new ItemExistsException(item.safeGetJCRPath());
498 	    } else {
499 		// there's already a node with that name
500 		// check same-name sibling setting of both new and existing node
501 		if (!def.allowSameNameSibs()
502 			|| !((NodeImpl) item).getDefinition().allowSameNameSibs()) {
503 		    throw new ItemExistsException(item.safeGetJCRPath());
504 		}
505 	    }
506 	} catch (PathNotFoundException pnfe) {
507 	    // no name collision
508 	}
509 
510 	// check if versioning allows write
511 	if (!safeIsCheckedOut()) {
512 	    String msg = safeGetJCRPath() + ": cannot add a child to a checked-in node";
513 	    log.error(msg);
514 	    throw new VersionException(msg);
515 	}
516 
517 	// check protected flag of parent (i.e. this) node
518 	if (getDefinition().isProtected()) {
519 	    String msg = safeGetJCRPath() + ": cannot add a child to a protected node";
520 	    log.error(msg);
521 	    throw new ConstraintViolationException(msg);
522 	}
523 
524 	// now do create the child node
525 	return createChildNode(nodeName, def, nodeType, uuid);
526     }

528     private void setMixinTypesProperty(Set mixinNames) throws RepositoryException {
529 	NodeState thisState = (NodeState) state;
530 	// get or create jcr:mixinTypes property
531 	PropertyImpl prop = null;
532 	if (thisState.hasPropertyEntry(PROPNAME_MIXINTYPES)) {
533 	    prop = (PropertyImpl) itemMgr.getItem(new PropertyId(thisState.getUUID(), PROPNAME_MIXINTYPES));
534 	} else {
535 	    // find definition for the jcr:mixinTypes property and create property
536 	    PropertyDefImpl def = getApplicablePropertyDef(PROPNAME_MIXINTYPES, PropertyType.NAME, true);
537 	    prop = createChildProperty(PROPNAME_MIXINTYPES, PropertyType.NAME, def);
538 	}
539 
540 	if (mixinNames.isEmpty()) {
541 	    // purge empty jcr:mixinTypes property
542 	    removeChildProperty(PROPNAME_MIXINTYPES);
543 	    return;
544 	}
545 
546 	// call internalSetValue for setting the jcr:mixinTypes property
547 	// to avoid checking of the 'protected' flag
548 	InternalValue[] vals = new InternalValue[mixinNames.size()];
549 	Iterator iter = mixinNames.iterator();
550 	int cnt = 0;
551 	while (iter.hasNext()) {
552 	    vals[cnt++] = InternalValue.create((QName) iter.next());
553 	}
554 	prop.internalSetValue(vals, PropertyType.NAME);
555     }

615 	if (!isTransient()) {

1158     public void orderBefore(String srcName, String destName)
1159 	    throws UnsupportedRepositoryOperationException, ConstraintViolationException,
1160 	    ItemNotFoundException, RepositoryException {
1161 	if (!nodeType.hasOrderableChildNodes()) {
1162 	    throw new UnsupportedRepositoryOperationException("child node ordering not supported on node " + safeGetJCRPath());
1163 	}
1164 
1165 	// check arguments
1166 	if (srcName.equals(destName)) {
1167 	    throw new ConstraintViolationException("source and destination have to be different");
1168 	}
1169 
1170 	Path.PathElement insertName;
1171 	try {
1172 	    Path p = Path.create(srcName, session.getNamespaceResolver(), false);
1173 	    if (p.getAncestorCount() > 0) {
1174 		throw new RepositoryException("invalid name: " + srcName);
1175 	    }
1176 	    insertName = p.getNameElement();
1177 	} catch (MalformedPathException e) {
1178 	    String msg = "invalid name: " + srcName;
1179 	    log.error(msg, e);
1180 	    throw new RepositoryException(msg, e);
1181 	}
1182 
1183 	Path.PathElement beforeName;
1184 	if (destName != null) {
1185 	    try {
1186 		Path p = Path.create(destName, session.getNamespaceResolver(), false);
1187 		if (p.getAncestorCount() > 0) {
1188 		    throw new RepositoryException("invalid name: " + destName);
1189 		}
1190 		beforeName = p.getNameElement();
1191 	    } catch (MalformedPathException e) {
1192 		String msg = "invalid name: " + destName;
1193 		log.error(msg, e);
1194 		throw new RepositoryException(msg, e);
1195 	    }
1196 	} else {
1197 	    beforeName = null;
1198 	}
1199 
1200 	// check existence
1201 	if (!hasNode(srcName)) {
1202 	    throw new ItemNotFoundException(safeGetJCRPath() + " has no child node with name " + srcName);
1203 	}
1204 	if (destName != null && !hasNode(destName)) {
1205 	    throw new ItemNotFoundException(safeGetJCRPath() + " has no child node with name " + destName);
1206 	}
1207 
1208 	// check if versioning allows write
1209 	if (!safeIsCheckedOut()) {
1210 	    String msg = safeGetJCRPath() + ": cannot change child node ordering of a checked-in node";
1211 	    log.error(msg);
1212 	    throw new VersionException(msg);
1213 	}
1214 
1215 	// check protected flag
1216 	if (getDefinition().isProtected()) {
1217 	    String msg = safeGetJCRPath() + ": cannot change child node ordering of a protected node";
1218 	    log.error(msg);
1219 	    throw new ConstraintViolationException(msg);
1220 	}
1221 
1222 	ArrayList list = new ArrayList(((NodeState) state).getChildNodeEntries());
1223 	int srcInd = -1, destInd = -1;
1224 	for (int i = 0; i < list.size(); i++) {
1225 	    NodeState.ChildNodeEntry entry = (NodeState.ChildNodeEntry) list.get(i);
1226 	    if (srcInd == -1) {
1227 		if (entry.getName().equals(insertName.getName())
1228 			&& (entry.getIndex() == insertName.getIndex()
1229 			|| insertName.getIndex() == 0 && entry.getIndex() == 1)) {
1230 		    srcInd = i;
1231 		}
1232 	    }
1233 	    if (destInd == -1 && beforeName != null) {
1234 		if (entry.getName().equals(beforeName.getName())
1235 			&& (entry.getIndex() == beforeName.getIndex()
1236 			|| beforeName.getIndex() == 0 && entry.getIndex() == 1)) {
1237 		    destInd = i;
1238 		    if (srcInd != -1) {
1239 			break;
1240 		    }
1241 		}
1242 	    } else {
1243 		if (srcInd != -1) {
1244 		    break;
1245 		}
1246 	    }
1247 	}
1248 
1249 	// check if resulting order would be different to current order
1250 	if (destInd == -1) {
1251 	    if (srcInd == list.size() - 1) {

1654 	    RepositoryException {

1656 	checkItemState();
1657 
1658 	// check if versioning allows write
1659 	if (!safeIsCheckedOut()) {
1660 	    String msg = safeGetJCRPath() + ": cannot add a mixin node type to a checked-in node";
1661 	    log.error(msg);
1662 	    throw new ConstraintViolationException(msg);
1663 	}
1664 
1665 	// check protected flag
1666 	if (definition.isProtected()) {
1667 	    String msg = safeGetJCRPath() + ": cannot add a mixin node type to a protected node";
1668 	    log.error(msg);
1669 	    throw new ConstraintViolationException(msg);
1670 	}
1671 
1672 	QName ntName;
1673 	try {
1674 	    ntName = QName.fromJCRName(mixinName, session.getNamespaceResolver());
1675 	} catch (IllegalNameException ine) {
1676 	    throw new RepositoryException("invalid mixin type name: " + mixinName, ine);
1677 	} catch (UnknownPrefixException upe) {
1678 	    throw new RepositoryException("invalid mixin type name: " + mixinName, upe);
1679 	}
1680 
1681 	NodeTypeManagerImpl ntMgr = session.getNodeTypeManager();
1682 	NodeTypeImpl mixin = ntMgr.getNodeType(ntName);
1683 	if (!mixin.isMixin()) {
1684 	    throw new RepositoryException(mixinName + ": not a mixin node type");
1685 	}
1686 	if (nodeType.isDerivedFrom(ntName)) {
1687 	    throw new RepositoryException(mixinName + ": already contained in primary node type");
1688 	}
1689 
1690 	// build effective node type of mixin's & primary type in order to detect conflicts
1691 	NodeTypeRegistry ntReg = ntMgr.getNodeTypeRegistry();
1692 	EffectiveNodeType entExisting;
1693 	try {
1694 	    // existing mixin's
1695 	    HashSet set = new HashSet(((NodeState) state).getMixinTypeNames());
1696 	    // primary type
1697 	    set.add(nodeType.getQName());
1698 	    // build effective node type representing primary type including existing mixin's
1699 	    entExisting = ntReg.buildEffectiveNodeType((QName[]) set.toArray(new QName[set.size()]));
1700 	    if (entExisting.includesNodeType(ntName)) {
1701 		throw new RepositoryException(mixinName + ": already contained in mixin types");
1702 	    }
1703 	    // add new mixin
1704 	    set.add(ntName);
1705 	    // try to build new effective node type (will throw in case of conflicts)
1706 	    ntReg.buildEffectiveNodeType((QName[]) set.toArray(new QName[set.size()]));
1707 	} catch (NodeTypeConflictException ntce) {
1708 	    throw new ConstraintViolationException(ntce.getMessage());
1709 	}
1710 
1711 	// do the actual modifications implied by the new mixin;
1712 	// try to revert the changes in case an excpetion occurs
1713 	try {
1714 	    // modify the state of this node
1715 	    NodeState thisState = (NodeState) getOrCreateTransientItemState();
1716 	    // add mixin name
1717 	    Set mixins = new HashSet(thisState.getMixinTypeNames());
1718 	    mixins.add(ntName);
1719 	    thisState.setMixinTypeNames(mixins);
1720 
1721 	    // set jcr:mixinTypes property
1722 	    setMixinTypesProperty(mixins);
1723 
1724 	    // add 'auto-create' properties defined in mixin type
1725 	    PropertyDef[] pda = mixin.getAutoCreatePropertyDefs();
1726 	    for (int i = 0; i < pda.length; i++) {
1727 		PropertyDefImpl pd = (PropertyDefImpl) pda[i];

1729 		// or existing mixin's
1730 		NodeTypeImpl declaringNT = (NodeTypeImpl) pd.getDeclaringNodeType();
1731 		if (!entExisting.includesNodeType(declaringNT.getQName())) {
1732 		    createChildProperty(pd.getQName(), pd.getRequiredType(), pd);
1733 		}
1734 	    }
1735 
1736 	    // recursively add 'auto-create' child nodes defined in mixin type
1737 	    NodeDef[] nda = mixin.getAutoCreateNodeDefs();
1738 	    for (int i = 0; i < nda.length; i++) {
1739 		NodeDefImpl nd = (NodeDefImpl) nda[i];
1740 		// make sure that the child node is not already defined by primary type
1741 		// or existing mixin's
1742 		NodeTypeImpl declaringNT = (NodeTypeImpl) nd.getDeclaringNodeType();
1743 		if (!entExisting.includesNodeType(declaringNT.getQName())) {
1744 		    createChildNode(nd.getQName(), nd, (NodeTypeImpl) nd.getDefaultPrimaryType(), null);
1745 		}
1746 	    }
1747 
1748 	    // check for special node types
1749 	    // todo centralize version history creation code (currently in NodeImpl.addMixin & ItemImpl.initVersionHistories
1750 	    if (ntName.equals(NodeTypeRegistry.MIX_VERSIONABLE)) {
1751 		VersionHistory hist = rep.getVersionManager().createVersionHistory(this);
1752 		internalSetProperty(VersionImpl.PROPNAME_VERSION_HISTORY, InternalValue.create(new UUID(hist.getUUID())));

1760 		removeMixin(mixinName);
1761 	    } catch (RepositoryException re1) {
1762 		// silently ignore & fall through
1763 	    }
1764 	    throw re;
1765 	}
1766     }

1802 	    throw new NoSuchNodeTypeException(mixinName);

1828 	NodeTypeManagerImpl ntMgr = session.getNodeTypeManager();
1829 	NodeTypeRegistry ntReg = ntMgr.getNodeTypeRegistry();
1830 	EffectiveNodeType entRemaining;

1837 	    entRemaining = ntReg.buildEffectiveNodeType((QName[]) set.toArray(new QName[set.size()]));
1838 	} catch (NodeTypeConflictException ntce) {

1848 	    // i.e. we're done

1856 	ArrayList tmp = new ArrayList(thisState.getPropertyEntries());

2349     /***
2350      * Performs the merge test. If the result is 'update', then the corresponding
2351      * source node is returned. if the result is 'leave' or 'besteffort-fail'
2352      * then <code>null</code> is returned. If the result of the merge test is
2353      * 'fail' with bestEffort set to <code>false</code> a MergeException is
2354      * thrown.
2355      *
2356      * @param srcWorkspace
2357      * @param bestEffort
2358      * @return
2359      * @throws RepositoryException
2360      * @throws AccessDeniedException
2361      */
2362     private NodeImpl doMergeTest(String srcWorkspace, boolean bestEffort)
2363 	    throws RepositoryException, AccessDeniedException {
2364 
2365 	// If N does not have a corresponding node then the merge result for N is leave.
2366 	NodeImpl srcNode = getCorrespondingNode(srcWorkspace);
2367 	if (srcNode == null) {
2368 	    return null;
2369 	}
2370 
2371 	// if not versionable, update
2372 	if (!isNodeType(NodeTypeRegistry.MIX_VERSIONABLE)) {
2373 	    return srcNode;
2374 	}
2375 	// if source node is not versionable, leave
2376 	if (!srcNode.isNodeType(NodeTypeRegistry.MIX_VERSIONABLE)) {
2377 	    return null;
2378 	}
2379 	// test versions
2380 	VersionImpl v = (VersionImpl) getBaseVersion();
2381 	VersionImpl vp = (VersionImpl) srcNode.getBaseVersion();
2382 	if (vp.isMoreRecent(v) && !isCheckedOut()) {
2383 	    // I f V' is a successor (to any degree) of V, then the merge result for
2384 	    // N is update. This case can be thought of as the case where N' is
2385 	    // newer than N and therefore N should be updated to reflect N'.
2386 	    return srcNode;
2387 	} else if (v.isSame(vp) || v.isMoreRecent(vp)) {
2388 	    // If V' is a predecessor (to any degree) of V or if V and V' are
2389 	    // identical (i.e., are actually the same version), then the merge
2390 	    // result for N is leave. This case can be thought of as the case where
2391 	    // N' is older or the same age as N and therefore N should be left alone.
2392 	    return null;
2393 	} else {
2394 	    // If V is neither a successor of, predecessor of, nor identical
2395 	    // with V', then the merge result for N is failed. This is the case
2396 	    // where N and N' represent divergent branches of the version graph,
2397 	    // thus determining the result of a merge is non-trivial.
2398 	    if (bestEffort) {
2399 		// add 'offending' version to jcr:mergeFailed property
2400 		if (hasProperty(ItemImpl.PROPNAME_MERGE_FAILED)) {
2401 		    Value[] values = getProperty(ItemImpl.PROPNAME_MERGE_FAILED).getValues();
2402 		    Value[] newValues = new Value[values.length + 1];
2403 		    System.arraycopy(values, 0, newValues, 0, values.length);
2404 		    newValues[values.length] = new ReferenceValue(vp);
2405 		    setProperty(ItemImpl.PROPNAME_MERGE_FAILED, newValues);
2406 		} else {
2407 		    Value[] newValues = new Value[1];
2408 		    newValues[0] = new ReferenceValue(vp);
2409 		    setProperty(ItemImpl.PROPNAME_MERGE_FAILED, newValues);
2410 		}
2411 		return null;
2412 	    } else {
2413 		String msg = "Unable to merge nodes. Violating versions. " + safeGetJCRPath();
2414 		log.debug(msg);
2415 		throw new MergeException(msg);
2416 	    }
2417 	}
2418     }
