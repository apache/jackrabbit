18  import java.io.ByteArrayOutputStream;
19  import java.util.BitSet;

21  /***
22   * The <code>FileSystemPathUtil</code> utility class ...

27  public final class FileSystemPathUtil {
28  
29      private static final char[] hexTable = "0123456789abcdef".toCharArray();
30  
31      private static final char ESCAPE_CHAR = '%';
32  
33      /***
34       * The list of characters that are not encoded by the <code>escapeName(String)</code>
35       * and <code>unescape(String)</code> methods. They contains the characters
36       * which can savely be used in file names:
37       */
38      public static final BitSet SAVE_NAMECHARS;
39  
40      /***
41       * The list of characters that are not encoded by the <code>escapePath(String)</code>
42       * and <code>unescape(String)</code> methods. They contains the characters
43       * which can savely be used in file paths:
44       */
45      public static final BitSet SAVE_PATHCHARS;
46  
47      static {
48  	// build list of valid name characters
49  	SAVE_NAMECHARS = new BitSet(256);
50  	int i;
51  	for (i = 'a'; i <= 'z'; i++) {
52  	    SAVE_NAMECHARS.set(i);
53  	}
54  	for (i = 'A'; i <= 'Z'; i++) {
55  	    SAVE_NAMECHARS.set(i);
56  	}
57  	for (i = '0'; i <= '9'; i++) {
58  	    SAVE_NAMECHARS.set(i);
59  	}
60  	SAVE_NAMECHARS.set('-');
61  	SAVE_NAMECHARS.set('_');
62  	SAVE_NAMECHARS.set('.');
63  
64  	// build list of valid path characters (inlcudes name characters)
65  	SAVE_PATHCHARS = (BitSet) SAVE_NAMECHARS.clone();
66  	SAVE_PATHCHARS.set(FileSystem.SEPARATOR_CHAR);
67      }
68  
69      /***
70       * private constructor
71       */
72      private FileSystemPathUtil() {
73      }
74  
75      private static String escape(String s, BitSet saveChars) {
76  	byte[] bytes = s.getBytes();
77  	StringBuffer out = new StringBuffer(bytes.length);
78  	for (int i = 0; i < bytes.length; i++) {
79  	    int c = bytes[i] & 0xff;
80  	    if (saveChars.get(c) && c != ESCAPE_CHAR) {
81  		out.append((char) c);
82  	    } else {
83  		out.append(ESCAPE_CHAR);
84  		out.append(hexTable[(c >> 4) & 0x0f]);
85  		out.append(hexTable[(c) & 0x0f]);
86  	    }
87  	}
88  	return out.toString();
89      }
90  
91      /***
92       * Encodes the specified <code>path</code>. Same as
93       * <code>{@link #escapeName(String)}</code> except that the separator
94       * character <b><code>/</code></b> is regarded as a legal path character
95       * that needs no escaping.
96       *
97       * @param path the path to encode.
98       * @return the escaped path
99       */
100     public static String escapePath(String path) {
101 	return escape(path, SAVE_PATHCHARS);
102     }
103 
104     /***
105      * Encodes the specified <code>name</code>. Same as
106      * <code>{@link #escapePath(String)}</code> except that the separator character
107      * <b><code>/</code></b> is regarded as an illegal character that needs
108      * escaping.
109      *
110      * @param name the name to encode.
111      * @return the escaped name
112      */
113     public static String escapeName(String name) {
114 	return escape(name, SAVE_NAMECHARS);
115     }
116 
117     /***
118      * Decodes the specified path/name.
119      *
120      * @param pathOrName the escaped path/name
121      * @return the unescaped path/name
122      */
123     public static String unescape(String pathOrName) {
124 	ByteArrayOutputStream out = new ByteArrayOutputStream(pathOrName.length());
125 	for (int i = 0; i < pathOrName.length(); i++) {
126 	    char c = pathOrName.charAt(i);
127 	    if (c == ESCAPE_CHAR) {
128 		try {
129 		    out.write(Integer.parseInt(pathOrName.substring(i + 1, i + 3), 16));
130 		} catch (NumberFormatException e) {
131 		    throw new IllegalArgumentException();
132 		}
133 		i += 2;
134 	    } else {
135 		out.write(c);
136 	    }
137 	}
138 	return new String(out.toByteArray());
139     }

141     /***
142      * Returns the parent directory of the specified <code>path</code>.
143      *
144      * @param path a file system path denoting a directory or a file.
145      * @return the parent directory.
146      */
147     public static String getParentDir(String path) {
148 	int pos = path.lastIndexOf(FileSystem.SEPARATOR_CHAR);
149 	if (pos > 0) {
150 	    return path.substring(0, pos);
151 	}
152 	return FileSystem.SEPARATOR;
153     }
154 
155     /***
156      * Returns the name of the specified <code>path</code>.
157      *
158      * @param path a file system path denoting a directory or a file.
159      * @return the name.
160      */
161     public static String getName(String path) {
162 	int pos = path.lastIndexOf(FileSystem.SEPARATOR_CHAR);
163 	if (pos != -1) {
164 	    return path.substring(pos + 1);
165 	}
166 	return path;
167     }
