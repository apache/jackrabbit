31  /***
32   * <code>ValueConstraint</code> and its subclasses are used to check the
33   * syntax of a value constraint and to test if a specific value satisfies
34   * it.
35   *
36   * @author Stefan Guggisberg
37   * @version $Revision: 1.15 $, $Date: 2004/08/25 16:44:49 $
38   */
39  public abstract class ValueConstraint {
40      protected static Logger log = Logger.getLogger(ValueConstraint.class);
41  
42      final String definition;
43  
44      protected ValueConstraint(String definition) {
45  	this.definition = definition;
46      }
47  
48      public String getDefinition() {
49  	return definition;
50      }
51  
52      public static ValueConstraint create(int type, String definition) throws InvalidConstraintException {
53  	if (definition == null) {
54  	    throw new IllegalArgumentException("illegal definition (null)");
55  	}
56  	switch (type) {
57  	    case PropertyType.STRING:
58  		return new StringConstraint(definition);
59  
60  	    case PropertyType.BOOLEAN:
61  		return new BooleanConstraint(definition);
62  
63  	    case PropertyType.BINARY:
64  		return new NumericConstraint(definition);
65  
66  	    case PropertyType.DATE:
67  		return new DateConstraint(definition);
68  
69  	    case PropertyType.LONG:
70  	    case PropertyType.DOUBLE:
71  		return new NumericConstraint(definition);
72  
73  	    case PropertyType.NAME:
74  		// @todo implement NAME value constraint
75  		//return new NameConstraint(definition, nsResolver);
76  		return new StringConstraint(".*");
77  
78  	    case PropertyType.PATH:
79  		// @todo implement PATH value constraint
80  		//return new PathConstraint(definition, nsResolver);
81  		return new StringConstraint(".*");
82  
83  	    case PropertyType.REFERENCE:
84  		// @todo implement REFERENCE value constraint
85  		//return new ReferenceConstraint(definition, ntReg, nsResolver);
86  		return new ReferenceConstraint(definition, null, null);
87  
88  	    default:
89  		throw new IllegalArgumentException("unknown/unsupported target type for constraint: " + PropertyType.nameFromValue(type));
90  	}
91      }
92  
93      abstract void check(InternalValue value) throws ConstraintViolationException, RepositoryException;
94  }

99  class BooleanConstraint extends ValueConstraint {
100     final boolean reqBool;
101 
102     BooleanConstraint(String definition) throws InvalidConstraintException {
103 	super(definition);
104 
105 	// constraint format: 'true' or 'false'
106 	if (definition.equals("true")) {
107 	    reqBool = true;
108 	} else if (definition.equals("false")) {
109 	    reqBool = false;
110 	} else {
111 	    String msg = "'" + definition + "' is not a valid value constraint format for BOOLEAN values";
112 	    log.error(msg);
113 	    throw new InvalidConstraintException(msg);
114 	}
115     }
116 
117     void check(Boolean bool) throws ConstraintViolationException {
118 	if (bool == null) {
119 	    throw new ConstraintViolationException("null value does not satisfy the constraint '" + definition + "'");
120 	}
121 	check(bool.booleanValue());
122     }
123 
124     void check(boolean bool) throws ConstraintViolationException {
125 	if (bool != reqBool) {
126 	    throw new ConstraintViolationException("'" + bool + "' does not satisfy the constraint '" + definition + "'");
127 	}
128     }
129 
130     void check(InternalValue value) throws ConstraintViolationException, RepositoryException {
131 	if (value == null) {
132 	    throw new ConstraintViolationException("null value does not satisfy the constraint '" + definition + "'");
133 	}
134 	switch (value.getType()) {
135 	    case PropertyType.BOOLEAN:
136 		check((Boolean) value.internalValue());
137 		return;
138 
139 	    default:
140 		String msg = "BOOLEAN constraint can not be applied to value of type: " + PropertyType.nameFromValue(value.getType());
141 		log.error(msg);
142 		throw new RepositoryException(msg);
143 	}
144     }
145 }

148  * <code>StringConstraint</code> ...
149  */
150 class StringConstraint extends ValueConstraint {
151     final Pattern pattern;
152 
153     StringConstraint(String definition) throws InvalidConstraintException {
154 	super(definition);
155 
156 	// constraint format: regexp
157 	try {
158 	    pattern = Pattern.compile(definition);
159 	} catch (PatternSyntaxException pse) {
160 	    String msg = "'" + definition + "' is not valid regular expression syntax";
161 	    log.error(msg, pse);
162 	    throw new InvalidConstraintException(msg, pse);
163 	}
164     }
165 
166     void check(String text) throws ConstraintViolationException {
167 	if (text == null) {
168 	    throw new ConstraintViolationException("null value does not satisfy the constraint '" + definition + "'");
169 	}
170 	Matcher matcher = pattern.matcher(text);
171 	if (!matcher.matches()) {
172 	    throw new ConstraintViolationException("'" + text + "' does not satisfy the constraint '" + definition + "'");
173 	}
174     }
175 
176     void check(InternalValue value) throws ConstraintViolationException, RepositoryException {
177 	if (value == null) {
178 	    throw new ConstraintViolationException("null value does not satisfy the constraint '" + definition + "'");
179 	}
180 	switch (value.getType()) {
181 	    case PropertyType.STRING:
182 		check(value.toString());
183 		return;
184 
185 	    default:
186 		String msg = "STRING constraint can not be applied to value of type: " + PropertyType.nameFromValue(value.getType());
187 		log.error(msg);
188 		throw new RepositoryException(msg);
189 	}
190     }

193 /***
194  * <code>NumericConstraint</code> ...
195  */
196 class NumericConstraint extends ValueConstraint {
197     final boolean lowerInclusive;
198     final Double lowerLimit;
199     final boolean upperInclusive;
200     final Double upperLimit;
201 
202     NumericConstraint(String definition) throws InvalidConstraintException {
203 	super(definition);
204 
205 	// format: '(<min>, <max>)',  '[<min>, <max>]', '(, <max>)' etc.
206 	Pattern pattern = Pattern.compile("([//(//[]) *(//-?//d+//.?//d*)? *, *(//-?//d+//.?//d*)? *([//)//]])");
207 	Matcher matcher = pattern.matcher(definition);
208 	if (matcher.matches()) {
209 	    try {
210 		// group 1 is lower inclusive/exclusive
211 		String s = matcher.group(1);
212 		lowerInclusive = s.equals("[");
213 		// group 2 is lower limit
214 		s = matcher.group(2);
215 		if (s == null || s.length() == 0) {
216 		    lowerLimit = null;
217 		} else {
218 		    lowerLimit = Double.valueOf(matcher.group(2));
219 		}
220 		// group 3 is upper limit
221 		s = matcher.group(3);
222 		if (s == null || s.length() == 0) {
223 		    upperLimit = null;
224 		} else {
225 		    upperLimit = Double.valueOf(matcher.group(3));
226 		}
227 		// group 4 is lower inclusive/exclusive
228 		s = matcher.group(4);
229 		upperInclusive = s.equals("]");
230 		if (lowerLimit == null && upperLimit == null) {
231 		    String msg = "'" + definition + "' is not a valid value constraint format for numeric types: neither lower- nor upper-limit specified";
232 		    log.error(msg);
233 		    throw new InvalidConstraintException(msg);
234 		}
235 		if (lowerLimit != null && upperLimit != null) {
236 		    if (lowerLimit.doubleValue() > upperLimit.doubleValue()) {
237 			String msg = "'" + definition + "' is not a valid value constraint format for numeric types: lower-limit exceeds upper-limit";
238 			log.error(msg);
239 			throw new InvalidConstraintException(msg);
240 		    }
241 		}
242 	    } catch (NumberFormatException nfe) {
243 		String msg = "'" + definition + "' is not a valid value constraint format for numeric types";
244 		log.error(msg, nfe);
245 		throw new InvalidConstraintException(msg, nfe);
246 	    }
247 	} else {
248 	    String msg = "'" + definition + "' is not a valid value constraint format for numeric values";
249 	    log.error(msg);
250 	    throw new InvalidConstraintException(msg);
251 	}
252     }
253 
254     void check(Double number) throws ConstraintViolationException {
255 	if (number == null) {
256 	    throw new ConstraintViolationException("null value does not satisfy the constraint '" + definition + "'");
257 	}
258 	check(number.doubleValue());
259     }
260 
261     void check(Long number) throws ConstraintViolationException {
262 	if (number == null) {
263 	    throw new ConstraintViolationException("null value does not satisfy the constraint '" + definition + "'");
264 	}
265 	check(number.doubleValue());
266     }
267 
268     void check(double number) throws ConstraintViolationException {
269 	if (lowerLimit != null) {
270 	    if (lowerInclusive) {
271 		if (number < lowerLimit.doubleValue()) {
272 		    throw new ConstraintViolationException(number + " does not satisfy the constraint '" + definition + "'");
273 		}
274 	    } else {
275 		if (number <= lowerLimit.doubleValue()) {
276 		    throw new ConstraintViolationException(number + " does not satisfy the constraint '" + definition + "'");
277 		}
278 	    }
279 	}
280 	if (upperLimit != null) {
281 	    if (upperInclusive) {
282 		if (number > upperLimit.doubleValue()) {
283 		    throw new ConstraintViolationException(number + " does not satisfy the constraint '" + definition + "'");
284 		}
285 	    } else {
286 		if (number >= upperLimit.doubleValue()) {
287 		    throw new ConstraintViolationException(number + " does not satisfy the constraint '" + definition + "'");
288 		}
289 	    }
290 	}
291     }
292 
293     void check(InternalValue value) throws ConstraintViolationException, RepositoryException {
294 	if (value == null) {
295 	    throw new ConstraintViolationException("null value does not satisfy the constraint '" + definition + "'");
296 	}
297 	switch (value.getType()) {
298 	    case PropertyType.LONG:
299 		check((Long) value.internalValue());
300 		break;
301 
302 	    case PropertyType.DOUBLE:
303 		check((Double) value.internalValue());
304 		return;
305 
306 	    case PropertyType.BINARY:
307 		BLOBFileValue blob = (BLOBFileValue) value.internalValue();
308 		long length = blob.getLength();
309 		if (length != -1) {
310 		    check(length);
311 		} else {
312 		    log.warn("failed to determine length of binary value");
313 		    return;
314 		}
315 
316 	    default:
317 		String msg = "numeric constraint can not be applied to value of type: " + PropertyType.nameFromValue(value.getType());
318 		log.error(msg);
319 		throw new RepositoryException(msg);
320 	}
321     }
322 }

327 class DateConstraint extends ValueConstraint {
328     final boolean lowerInclusive;
329     final Calendar lowerLimit;
330     final boolean upperInclusive;
331     final Calendar upperLimit;
332 
333     DateConstraint(String definition) throws InvalidConstraintException {
334 	super(definition);
335 
336 	// format: '(<fromDate>, <toDate>)', '[<fromDate>, <toDate>]', '[, <toDate>]' etc.
337 	Pattern pattern = Pattern.compile("([//(//[]) *([0-9TZ//.//+-:]*)? *, *([0-9TZ//.//+-:]*)? *([//)//]])");
338 	Matcher matcher = pattern.matcher(definition);
339 	if (matcher.matches()) {
340 	    try {
341 		// group 1 is lower inclusive/exclusive
342 		String s = matcher.group(1);
343 		lowerInclusive = s.equals("[");
344 		// group 2 is lower limit
345 		s = matcher.group(2);
346 		if (s == null || s.length() == 0) {
347 		    lowerLimit = null;
348 		} else {
349 		    lowerLimit = DateValue.valueOf(matcher.group(2)).getDate();
350 		}
351 		// group 3 is upper limit
352 		s = matcher.group(3);
353 		if (s == null || s.length() == 0) {
354 		    upperLimit = null;
355 		} else {
356 		    upperLimit = DateValue.valueOf(matcher.group(3)).getDate();
357 		}
358 		// group 4 is upepr inclusive/exclusive
359 		s = matcher.group(4);
360 		upperInclusive = s.equals("]");
361 
362 		if (lowerLimit == null && upperLimit == null) {
363 		    String msg = "'" + definition + "' is not a valid value constraint format for dates: neither min- nor max-date specified";
364 		    log.error(msg);
365 		    throw new InvalidConstraintException(msg);
366 		}
367 		if (lowerLimit != null && upperLimit != null) {
368 		    if (lowerLimit.after(upperLimit)) {
369 			String msg = "'" + definition + "' is not a valid value constraint format for dates: min-date > max-date";
370 			log.error(msg);
371 			throw new InvalidConstraintException(msg);
372 		    }
373 		}
374 	    } catch (ValueFormatException vfe) {
375 		String msg = "'" + definition + "' is not a valid value constraint format for dates";
376 		log.error(msg, vfe);
377 		throw new InvalidConstraintException(msg, vfe);
378 	    } catch (RepositoryException re) {
379 		String msg = "'" + definition + "' is not a valid value constraint format for dates";
380 		log.error(msg, re);
381 		throw new InvalidConstraintException(msg, re);
382 	    }
383 	} else {
384 	    String msg = "'" + definition + "' is not a valid value constraint format for dates";
385 	    log.error(msg);
386 	    throw new InvalidConstraintException(msg);
387 	}
388     }
389 
390     void check(Calendar cal) throws ConstraintViolationException {
391 	if (cal == null) {
392 	    throw new ConstraintViolationException("null value does not satisfy the constraint '" + definition + "'");
393 	}
394 	if (lowerLimit != null) {
395 	    if (lowerInclusive) {
396 		if (cal.getTimeInMillis() < lowerLimit.getTimeInMillis()) {
397 		    throw new ConstraintViolationException(cal + " does not satisfy the constraint '" + definition + "'");
398 		}
399 	    } else {
400 		if (cal.getTimeInMillis() <= lowerLimit.getTimeInMillis()) {
401 		    throw new ConstraintViolationException(cal + " does not satisfy the constraint '" + definition + "'");
402 		}
403 	    }
404 	}
405 	if (upperLimit != null) {
406 	    if (upperInclusive) {
407 		if (cal.getTimeInMillis() > upperLimit.getTimeInMillis()) {
408 		    throw new ConstraintViolationException(cal + " does not satisfy the constraint '" + definition + "'");
409 		}
410 	    } else {
411 		if (cal.getTimeInMillis() >= upperLimit.getTimeInMillis()) {
412 		    throw new ConstraintViolationException(cal + " does not satisfy the constraint '" + definition + "'");
413 		}
414 	    }
415 	}
416     }
417 
418     void check(InternalValue value) throws ConstraintViolationException, RepositoryException {
419 	if (value == null) {
420 	    throw new ConstraintViolationException("null value does not satisfy the constraint '" + definition + "'");
421 	}
422 	switch (value.getType()) {
423 	    case PropertyType.DATE:
424 		check((Calendar) value.internalValue());
425 		return;
426 
427 	    default:
428 		String msg = "DATE constraint can not be applied to value of type: " + PropertyType.nameFromValue(value.getType());
429 		log.error(msg);
430 		throw new RepositoryException(msg);
431 	}
432     }
