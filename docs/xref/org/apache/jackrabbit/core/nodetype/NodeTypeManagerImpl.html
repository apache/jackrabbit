55      /***
56       * Constructor.
57       */
58      public NodeTypeManagerImpl(NodeTypeRegistry ntReg, NamespaceResolver nsResolver) {
59  	this.nsResolver = nsResolver;
60  	this.ntReg = ntReg;
61  	this.ntReg.addListener(this);
62  
63  	// setup item cache with soft references to node type instances
64  	ntCache = new ReferenceMap(ReferenceMap.HARD, ReferenceMap.SOFT);
65  
66  	rootNodeDef = new RootNodeDefinition(ntReg.getRootNodeDef(), this, nsResolver);
67      }

100     /***
101      * @param name
102      * @return
103      * @throws NoSuchNodeTypeException
104      */
105     public synchronized NodeTypeImpl getNodeType(QName name) throws NoSuchNodeTypeException {
106 	NodeTypeImpl nt = (NodeTypeImpl) ntCache.get(name);
107 	if (nt != null) {
108 	    return nt;
109 	}
110 
111 	EffectiveNodeType ent = ntReg.getEffectiveNodeType(name);
112 	NodeTypeDef def = ntReg.getNodeTypeDef(name);
113 	nt = new NodeTypeImpl(ent, def, this, nsResolver);
114 	ntCache.put(name, nt);
115 
116 	return nt;
117     }

142     //------------------------------------------------------< NodeTypeManager >
143     /***
144      * @see NodeTypeManager#getAllNodeTypes
145      */
146     public NodeTypeIterator getAllNodeTypes() throws RepositoryException {
147 	QName[] ntNames = ntReg.getRegisteredNodeTypes();
148 	ArrayList list = new ArrayList(ntNames.length);
149 	for (int i = 0; i < ntNames.length; i++) {
150 	    list.add(getNodeType(ntNames[i]));
151 	}
152 	return new IteratorHelper(Collections.unmodifiableCollection(list));
153     }
154 
155     /***
156      * @see NodeTypeManager#getPrimaryNodeTypes
157      */
158     public NodeTypeIterator getPrimaryNodeTypes() throws RepositoryException {
159 	QName[] ntNames = ntReg.getRegisteredNodeTypes();
160 	ArrayList list = new ArrayList(ntNames.length);
161 	for (int i = 0; i < ntNames.length; i++) {
162 	    NodeType nt = getNodeType(ntNames[i]);
163 	    if (!nt.isMixin()) {
164 		list.add(nt);
165 	    }
166 	}
167 	return new IteratorHelper(Collections.unmodifiableCollection(list));
168     }
169 
170     /***
171      * @see NodeTypeManager#getMixinNodeTypes
172      */
173     public NodeTypeIterator getMixinNodeTypes() throws RepositoryException {
174 	QName[] ntNames = ntReg.getRegisteredNodeTypes();
175 	ArrayList list = new ArrayList(ntNames.length);
176 	for (int i = 0; i < ntNames.length; i++) {
177 	    NodeType nt = getNodeType(ntNames[i]);
178 	    if (nt.isMixin()) {
179 		list.add(nt);
180 	    }
181 	}
182 	return new IteratorHelper(Collections.unmodifiableCollection(list));
183     }

185     /***
186      * @see NodeTypeManager#getNodeType
187      */
188     public NodeType getNodeType(String nodeTypeName) throws NoSuchNodeTypeException {
189 	try {
190 	    return getNodeType(QName.fromJCRName(nodeTypeName, nsResolver));
191 	} catch (UnknownPrefixException upe) {
192 	    throw new NoSuchNodeTypeException(nodeTypeName, upe);
193 	} catch (IllegalNameException ine) {
194 	    throw new NoSuchNodeTypeException(nodeTypeName, ine);
195 	}
196     }
